Read Only Properties

  • @property methods are 'getters', and the @x.setters/@y.setters are 'setters': 
# point.py
class WritecoordinateError(Exception):
    pass
    
    
class Point:
    def __init__(self, x, y):
        self._x = x
        self._y = y
        
    @property
    def x(self):
        return self._x
    
    @x.setter(self, value):
        raise WritecoordinateError("x coordinate is read-only")
        
    @property
    def y(self):
        return self._y
        
    @y.setter(self, value):
        raise WritecoordinateError("y coordinate is read-only")
