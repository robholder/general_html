Script - to take object data name (orange triangle) and rename the mesh data name to match (green triangle)

import bpy

def rename_mesh_data(collection):
    for obj in collection.objects:
        if obj.type == 'MESH':
            obj.data.name = obj.name
    for child_collection in collection.children:
        rename_mesh_data(child_collection)

rename_mesh_data(bpy.context.scene.collection)

Script - to apply all modifiers in a hierarchy

import bpy

def apply_all_modifiers():
    for obj in bpy.context.scene.objects:
        if obj.type == 'MESH':
            bpy.context.view_layer.objects.active = obj
            for modifier in obj.modifiers:
                bpy.ops.object.modifier_apply(modifier=modifier.name)

apply_all_modifiers()
