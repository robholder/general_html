Hacking With Swift - 244 - Challenge 3 - Sorting Options

Challenge 3

  • Allow users to customize the way contacts are sorted - by name or by most recent.

Adding Sort Options

Here is the complete, comprehensive guide to implementing dynamic sorting in SwiftData using SwiftUI’s modern view hierarchy.

Architectural Breakdown

When working with SwiftData's @Query macro, the sorting configurations are established when the property is compiled. If you attempt to dynamically change a sort descriptor inside the exact same view rendering your list, SwiftUI will not re-run the container's query framework.

To solve this, we separate our structure into a clean Parent-Child Architecture:

  • The Parent View (ProspectsView): Acts as the true Source of Truth. It owns the raw layout state (@State) tracking whether the user is editing, what items are selected, and what sorting rules are applied.
  • The Child View (ProspectsListView): Acts as the Data Engine. It intercepts the sort rules from the parent during initialization, dynamically runs the SwiftData @Query, and renders the interactive rows cleanly.

The Modern State Transmission System

Instead of passing states through long initializers using manual parameters, we use the modern Environment Key Architecture (introduced in iOS 17+).

The parent view injects its local @State bindings directly down into the view hierarchy bubble using custom .environment() modifiers. The child view then reads those bindings implicitly using @Environment. This keeps both codebases perfectly synchronized without creating broken, disconnected data copies.

Complete Project Implementation

To implement this setup in your project, create (or update) the following two source files.

1. ProspectsListView.swift

This dedicated child view isolates the @Query configuration and manages the interactive row items. 

import SwiftData
import SwiftUI

struct ProspectsListView: View {
    @Environment(\.modelContext) var modelContext
    
    // The query framework that dynamically fetches and reflects our models
    @Query var prospects: [Prospect]
    
    // Modern SwiftUI: Extract the data bindings straight out of the environment bubble
    @Environment(\.prospectsEditing) private var isEditing
    @Environment(\.prospectsSelection) private var selectedProspects
    
    var body: some View {
        List(prospects, selection: selectedProspects) { prospect in
            
            // CHALLENGE TIP: Using the simple form of NavigationLink to isolate touch hashes
            NavigationLink {
                EditProspectView(prospect: prospect)
            } label: {
                HStack {
                    VStack(alignment: .leading) {
                        Text(prospect.name)
                            .font(.headline)
                        Text(prospect.emailAddress)
                            .foregroundStyle(.secondary)
                        }
                    Spacer()
                    Image(
                        systemName: prospect.isContacted ?
                        "person.crop.circle.fill.badge.checkmark" :
                        "person.crop.circle.badge.xmark"
                    )
                    .font(.title2)
                    .foregroundStyle(prospect.isContacted ? .blue : .red)
                }
            }
            .swipeActions {
                Button("Delete", systemImage: "trash", role: .destructive) {
                    modelContext.delete(prospect)
                }
                if prospect.isContacted {
                    Button("Mark Uncontacted", systemImage: "person.crop.circle.badge.xmark") {
                        prospect.isContacted.toggle()
                    }
                    .tint(.blue)
                } else {
                    Button("Mark Contacted", systemImage: "person.crop.circle.fill.badge.checkmark") {
                        prospect.isContacted.toggle()
                    }
                    .tint(.green)
                    
                    Button("Remind Me", systemImage: "bell") {
                        addNotification(for: prospect)
                    }
                    .tint(.orange)
                }
            }
            .tag(prospect)
            
            // CHALLENGE MODIFIER: Bypasses simple row hyperlink taps while multi-selection is active
            .allowsHitTesting(!isEditing.wrappedValue)
        }
        // Force the List's underlying selection circles to reveal or hide natively
        .environment(\.editMode, .constant(isEditing.wrappedValue ? .active : .inactive))
    }
    
    /// DYNAMIC QUERY INITIALIZATION
    /// Every single time the parent passes a new activeSortDescriptor, SwiftUI rebuilds this child struct.
    /// This initializer intercepts the instruction and updates SwiftData's query engine before rendering.
    init(filter: ProspectsView.FilterType, sortOrder: SortDescriptor<Prospect>) {
        let showContactedOnly = filter == .contacted
        
        if filter == .none {
            // Apply sorting across the complete collection
            _prospects = Query(sort: [sortOrder])
        } else {
            // Apply filtering logic alongside the new sort rule simultaneously
            _prospects = Query(filter: #Predicate<Prospect> { prospect in
                prospect.isContacted == showContactedOnly
            }, sort: [sortOrder])
        }
    }
    
    // Notification scheduler logic helper
    func addNotification(for prospect: Prospect) {
        let center = UNUserNotificationCenter.current()
        let addRequest = {
            let content = UNMutableNotificationContent()
            content.title = "Contact \(prospect.name)"
            content.subtitle = prospect.emailAddress
            content.sound = UNNotificationSound.default
            
            let trigger = UNTimeIntervalNotificationTrigger(timeInterval: 5, repeats: false)
            let request = UNNotificationRequest(identifier: UUID().uuidString, content: content, trigger: trigger)
            center.add(request)
        }
        
        center.getNotificationSettings { settings in
            if settings.authorizationStatus == .authorized {
                addRequest()
            } else {
                center.requestAuthorization(options: [.alert, .badge, .sound]) { success, _ in
                    if success { addRequest() }
                }
            }
        }
    }
}

2. ProspectsView.swift

This parent view handles your global toolbar controls, core user states, and the dropdown sorting menu interface. 

import AVFoundation
import CodeScanner
import SwiftData
import SwiftUI

struct ProspectsView: View {
    @Environment(\.modelContext) var modelContext
    
    @State private var isShowingScanner = false
    @State private var selectedProspects = Set<Prospect>()
    @State private var isEditing = false
    
    // DYNAMIC SORT STATE: Holds the active parameters available for sorting
    enum SortOption { case name, email }
    @State private var currentSort: SortOption = .name
    
    enum FilterType { case none, contacted, uncontacted }
    let filter: FilterType
    
    var title: String {
        switch filter {
        case .none: "Everyone"
        case .contacted: "Contacted people"
        case .uncontacted: "Uncontacted people"
        }
    }
    
    // COMPUTED PROP: Dynamically generates a brand new SortDescriptor whenever currentSort switches
    var activeSortDescriptor: SortDescriptor<Prospect> {
        currentSort == .name ? SortDescriptor(\Prospect.name) : SortDescriptor(\Prospect.emailAddress)
    }
    
    var body: some View {
        NavigationStack {
            // Instantiate the child view and pass down the computed sort rule
            ProspectsListView(filter: filter, sortOrder: activeSortDescriptor)
                .navigationTitle(title)
                
                // DATA LINKAGE: Inject our active states seamlessly into the lower layout environment
                .environment(\.prospectsEditing, $isEditing)
                .environment(\.prospectsSelection, $selectedProspects)
                .toolbar {
                    // Left Group (Controls Selection Mode)
                    ToolbarItem(placement: .topBarLeading) {
                        HStack(spacing: 12) {
                            Button("Edit") {
                                withAnimation {
                                    isEditing.toggle()
                                    if !isEditing { selectedProspects.removeAll() }
                                }
                            }
                            .tint(isEditing ? .red : .blue)
                            
                            if isEditing && !selectedProspects.isEmpty {
                                Button(role: .destructive, action: deleteSelected) {
                                    Label("Delete Selection", systemImage: "trash.fill")
                                        .foregroundStyle(.red)
                                }
                            }
                        }
                    }
                    
                    // Right Group (Controls Actions & Menus)
                    ToolbarItem(placement: .topBarTrailing) {
                        HStack(spacing: 15) {
                            
                            // SORT OPTION DROPDOWN MENU
                            Menu {
                                Button {
                                    currentSort = .name
                                } label: {
                                    HStack {
                                        Text("Sort by Name")
                                        if currentSort == .name { Image(systemName: "checkmark") }
                                    }
                                }
                                
                                Button {
                                    currentSort = .email
                                } label: {
                                    HStack {
                                        Text("Sort by Email")
                                        if currentSort == .email { Image(systemName: "checkmark") }
                                    }
                                }
                            } label: {
                                // System design standard icon representing a sort action
                                Image(systemName: "arrow.up.and.down.text.horizontal")
                            }
                            
                            Button("Scan", systemImage: "qrcode.viewfinder") {
                                isShowingScanner = true
                            }
                        }
                    }
                }
                .sheet(isPresented: $isShowingScanner) {
                    CodeScannerView(codeTypes: [.qr], simulatedData: "Paul Hudson\npaul@hackingwithswift.com", completion: handleScan)
                }
        }
    }
    
    init(filter: FilterType) {
        self.filter = filter
    }
    
    func handleScan(result: Result<ScanResult, ScanError>) {
        isShowingScanner = false
        switch result {
        case .success(let result):
            let details = result.string.components(separatedBy: "\n")
            guard details.count == 2 else { return }
            let person = Prospect(name: details[0], emailAddress: details[1], isContacted: false)
            modelContext.insert(person)
        case .failure(let error):
            print("Scanning failed: \(error.localizedDescription)")
        }
    }
    
    func deleteSelected() {
        withAnimation {
            for prospect in selectedProspects {
                modelContext.delete(prospect)
            }
            selectedProspects.removeAll()
            isEditing = false
        }
    }
}

// MARK: - ENVIRONMENT REGISTER EXTRA (Place safely at the bottom of file or in an EnvironmentKeys.swift file)

struct ProspectsEditingKey: EnvironmentKey {
    static let defaultValue: Binding<Bool> = .constant(false)
}

struct ProspectsSelectionKey: EnvironmentKey {
    static let defaultValue: Binding<Set<Prospect>> = .constant([])
}

extension EnvironmentValues {
    var prospectsEditing: Binding<Bool> {
        get { self[ProspectsEditingKey.self] }
        set { self[ProspectsEditingKey.self] = newValue }
    }
    
    var prospectsSelection: Binding<Set<Prospect>> {
        get { self[ProspectsSelectionKey.self] }
        set { self[ProspectsSelectionKey.self] = newValue }
    }
}
