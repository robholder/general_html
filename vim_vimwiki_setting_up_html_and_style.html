Vimrc

This block includes the general setup for a pair of vimwikis, and includes the various settings to get the HTML converter to behave and cooperate with CSS styles and to find the 'template' file (template.tpl): 

"" setup for multiple wikis:
let wiki_1 = {}
let wiki_1.path = '~/vimwiki_dir/general/'
let wiki_1.path_html = '~/vimwiki_dir/general_html/'
let wiki_1.template_path = '~/vimwiki_dir/general/'
let wiki_1.template_default = 'template'

let wiki_2 = {}
let wiki_2.path = '~/vimwiki_dir/alt/'
let wiki_2.path_html = '~/vimwiki_dir/alt_html/'
let wiki_2.template_path = '~/vimwiki_dir/alt/'
let wiki_2.template_default = 'template'

let g:vimwiki_list = [wiki_1, wiki_2]
"
let g:vimwiki_aliases_file = expand("$HOME/vimwiki_aliases.wiki")
"
" Disable swap files for vimwiki pages:
autocmd FileType vimwiki setlocal noswapfile
"
"
""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
" Disable Vimwiki's built-in syntax highlighting (for prism css) let g:vimwiki_hl_headers = 0
let g:vimwiki_hl_cb_checked = 0
let g:vimwiki_hl_cb_unchecked = 0
let g:vimwiki_hl_code = 0
"
let g:vimwiki_valid_html_tags = '.*'
let g:vimwiki_disable_html_pre = 1
let g:vimwiki_use_mouse = 0
"
" Disable Vimwiki HTML syntax highlighting
let g:vimwiki_html_disable_pretty = 1
let g:vimwiki_html_disable_syntax = 1
"

<!-- Fixer script: converts <pre swift> to Prism format -->
<script>
document.addEventListener("DOMContentLoaded", () => {
  document.querySelectorAll("pre").forEach(pre => {
    // 1. Get the language (handles <pre swift> or <pre class="python">)
    //let lang = pre.getAttributeNames().find(a => a !== "class") || "text";
    let lang = pre.getAttributeNames().find(a => !["class", "style", "id"].includes(a)) || "text";
    
    // 2. Clean the pre tag
    pre.className = "language-" + lang;
    
    // 3. Wrap content in code tag if not already there
    if (!pre.querySelector("code")) {
      const code = document.createElement("code");
      code.className = "language-" + lang;
      code.innerHTML = pre.innerHTML;
      pre.innerHTML = "";
      pre.appendChild(code);
    }
  });
  Prism.highlightAll();
});
</script>

Style CSS File - style.css

This is a file called style.css and is stored with all the vimwiki HTML files.

This will be referenced by the template.tpl file (shown above):
