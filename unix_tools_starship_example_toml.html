Used with a zshrc setup block

#------------------------------------------------------------------------------#
# 3. STARSHIP CONFIGURATION (Hostname & Palette Logic)
#------------------------------------------------------------------------------#
#-------------------------------- Add Starship --------------------------------#
## 1. ORIGINAL Set config from default of ~/.config to ~/.config/starship/
#export STARSHIP_CONFIG=~/.config/starship/starship.toml
## 2. ORIGINAL Start Starship
#eval "$(starship init zsh)"
#
# 1. Path to your master config file, and other paths and defaults
SOURCE_TOML="$HOME/.config/starship/starship.toml"
TEMP_CONFIG="/tmp/starship_rendered.toml"
TARGET_PALETTE="default" 

# 1. Maintain your fleet here (Comments are ignored by Zsh)
KNOWN_HOSTS=(
    "macbook-m3"      # MacBook Pro - 2023 - M3 (Blue Accent)
    "macbook-intel"   # MacBook Pro - 2012 - Intel (Purple Accent)
    "macmini"         # MacMini - 2020 (Green Accent)
)

# 2. Setup paths and defaults
#  Get host (Detects OS automatically - Linux/MacOS )
if [[ "$OSTYPE" == "darwin"* ]]; then
    ## Simple (slower?) way
    #CURRENT_HOST=$(hostname -s) # macOS
    # macOS: The %m expansion is the internal Zsh short hostname - faster
    CURRENT_HOST=${(%):-%m}
else
    # Simple (less reliable with long host 'dot' hostnames
    #CURRENT_HOST=$(hostname)    # Linux
    # Linux: Fallback to standard hostname
    CURRENT_HOST=$(hostname | cut -d. -f1)
fi

# 3. Check if CURRENT_HOST exists in the array
# This specific Zsh syntax (i) returns the index if found
if [[ ${KNOWN_HOSTS[(i)$CURRENT_HOST]} -le ${#KNOWN_HOSTS} ]]; then
    TARGET_PALETTE="$CURRENT_HOST"
fi

# 4. Inject the palette and fire up Starship
# We check if the source file exists first to avoid sed errors
if [[ -f "$SOURCE_TOML" ]]; then
    sed "s/SUBSTITUTE_MACHINE_NAME/$TARGET_PALETTE/" "$SOURCE_TOML" > "$TEMP_CONFIG"
    export STARSHIP_CONFIG="$TEMP_CONFIG"
fi

# 4. Start Starship 
export STARSHIP_CONFIG="$TEMP_CONFIG"
eval "$(starship init zsh)"
#------------------------------------ END -------------------------------------#
