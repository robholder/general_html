Hacking With Swift - 165 - Challenge 2 - Adding a sort menu

Adding a Sort Menu to the iExpense App

Starting state

ContentView.swift

import SwiftData
import SwiftUI

@Model
class ExpenseItem {
    
    var name: String
    var type: String
    var amount: Double
    
    init(name: String, type: String, amount: Double) {
        self.name = name
        self.type = type
        self.amount = amount
    }
}


struct ExpenseRow: View {
    let item: ExpenseItem
    
    var amountColor: Color {
        switch item.amount {
        case ..<10: return .primary
        case ..<100: return .green
        default: return .red
        }
    }
    
    var body: some View {
        HStack {
            VStack(alignment: .leading) {
                Text(item.name)
                    .font(.headline)
                Text(item.type)
                    .font(.subheadline)
                    .foregroundStyle(.secondary)
            }
            
            Spacer()
            
            Text(item.amount, format: .currency(code: Locale.current.currency?.identifier ?? "USD"))
                .foregroundStyle(amountColor)
        }
    }
}

struct ContentView: View {
    @Environment(\.modelContext) var modelContext
    @Query var items: [ExpenseItem]
    
    var personalExpenses: [ExpenseItem] {
        items.filter { $0.type == "Personal" }
    }
    var businessExpenses: [ExpenseItem] {
        items.filter { $0.type == "Business" }
    }
    
    var body: some View {
        NavigationStack {
            List {
                Section("Personal") {
                    ForEach(personalExpenses) { item in
                        ExpenseRow(item: item)
                    }
                    .onDelete { offsets in
                        removeItems(at: offsets, in: personalExpenses)
                    }
                }
                
                Section("Business") {
                    ForEach(businessExpenses) { item in
                        ExpenseRow(item: item)
                    }
                    .onDelete { offsets in
                        removeItems(at: offsets, in: businessExpenses)
                    }
                }
            }
            .navigationTitle("iExpense")
            .toolbar {
                NavigationLink {
                    AddView()
                } label: {
                    Label("Add Expense", systemImage: "plus")
                }
            }
        }
    }
    
    func removeItems(at offsets: IndexSet, in source: [ExpenseItem]) {
        for offset in offsets {
            let item = source[offset]
            modelContext.delete(item)
        }
    }
}

#Preview {
    ContentView()
}

Adding the Sort Functionality

We Need a New ExpenseListView

We need to extract the List from ContentView to a new ExpenseListView to control the sorting facility.

Here is a conceptual guide to understanding why we have to refactor the view, followed by a step-by-step strategic checklist you can reuse in future projects.

The Core Strategy: Why Two Views?

The problem boils down to a strict limitation in SwiftUI: @Query properties are locked in place once a View is created. They cannot listen to changes from a @State variable inside the same struct.

To bypass this roadblock, we use an architectural pattern called View Extraction.

By splitting the code into a Parent View (ContentView) and a Child View (ExpenseListView), we change the data flow:

  • The Parent tracks what the user clicked using @State.
  • When the user changes the sort picker, the Parent forces the Child to completely destroy and recreate itself.
  • During recreation, the Child's custom init() function interceptively rebuilds the @Query with the updated rules before any data hits the screen.

Step-by-Step Refactoring Checklist

Whenever you need to add dynamic sorting or filtering to a SwiftData @Query, use this exact 4-step workflow:

Step 1: Create the Child Struct

Create an entirely new View structure. Cut the data-dependent UI elements (usually the List) out of your main view and paste them into this new subview.

Step 2: Define the Child's Custom Initializer

Instead of declaring a standard @Query, you leave the query uninitialized. You use a custom init method to receive the sorting rule from the parent and inject it straight into the underlying property wrapper using the underscore (_items) syntax. 

struct MyListView: View {
    @Query var items: [ExpenseItem] // 1. Uninitialized query
    
    init(sortOrder: [SortDescriptor<ExpenseItem>]) {
        // 2. Underscore targets the Query wrapper itself, configuring it dynamically
        _items = Query(sort: sortOrder) 
    }
    // ... UI goes here
}

Step 3: Set Up Parent State and Menu UI

Inside your main ContentView, create a @State variable containing a default SortDescriptor. Build your Menu and Picker in the toolbar, assigning corresponding array tags to match your data types. 

@State private var sortOrder = [SortDescriptor(\ExpenseItem.name)]

// Picker Items use arrays matching the State type:
Text("Sort by Amount").tag([SortDescriptor(\ExpenseItem.amount)])

Step 4: Pass State from Parent to Child

Finally, drop your newly created child view into the body of your parent view, passing the @State variable forward as a normal parameter. 

var body: some View {
    NavigationStack {
        MyListView(sortOrder: sortOrder) // State automatically flows downward
    }
}

Why this is highly repeatable

This layout scales cleanly. If you later decide to add a Search Bar or a Category Filter, you do not need to rewrite your views again. You simply expand the child view's initializer to accept a filter string or predicate alongside the sorting rules, allowing ContentView to remain a lightweight control dashboard while the subview handles the database heavy lifting.
