In Context

import CoreImage
import CoreImage.CIFilterBuiltins
import PhotosUI
import StoreKit
import SwiftUI

struct ContentView: View {
    @State private var selectedItem: PhotosPickerItem?
    @State private var processedImage: Image?
    
    @State private var filterIntensity = 0.5
    // Added to track if intensity key is present on selected filter
    @State private var intensityPresent = false
    @State private var filterRadius = 0.5
    // Added to track if radius key is present on selected filter
    @State private var radiusPresent = false
    @State private var filterScale = 0.5
    // Added to track if scale key is present on selected filter
    @State private var scalePresent = false

    // We need a context to convert a CIImage -> CGImage
    // and a filter to apply
    // Core Image Context is an object for rendering a CIImage to a CGImage
    // Recipe -> image
    let context = CIContext()
    // Place holder to update:
    //@State private var currentFilter = CIFilter.sepiaTone()
    // Updated to set type to CIFilter, rather then CIFilter.sepiaTone only.
    // Now we lose acdess to the 'intensity' property.
    @State private var currentFilter: CIFilter = CIFilter.sepiaTone()
    
    // Keep beginImage in a var to prevent having to re-select everytime
    // the filter is selected.
    //@State private var beginImage: CIImage?
    
    @State private var showingFilters = false
    
    // Counter to count how many times the user is changing filter
    @AppStorage("filterCount") var filterCount = 0
    // Set up Request Review environment
    @Environment(\.requestReview) var requestReview

    var body: some View {
        NavigationStack {
            VStack {
                Spacer()
                
                // Wrap image secion with picker closure
                PhotosPicker(selection: $selectedItem) {
                    // Unwrap optional (processedImage)
                    if let processedImage {
                        processedImage
                            .resizable()
                            .scaledToFit()
                    } else {
                        // Show 'unavailable' screen if no image selected
                        ContentUnavailableView(
                            "No Picture",
                            systemImage: "photo.badge.plus",
                            description: Text("Tap to import a photo")
                        )
                    }
                }
                // Stop the 'No Picture' view from turning active blue
                .buttonStyle(.plain)
                .onChange(of: selectedItem, loadImage)
                
                Spacer()
                
                HStack {
                    // Add slider control (filter intensity)
                    Text("Intensity")
                    Slider(value: $filterIntensity)
                    // Add 'onChange' to watch for live changes of intensity slider
                        .onChange(of: filterIntensity, applyProcessing)
                        // Disable button if parameter not present in Keys
                        .disabled(!intensityPresent)
                }
                // Reduce visibility if parameter not present in Keys
                .opacity(!intensityPresent ? 0.3 : 1)
                HStack {
                    // Add slider control (filter radius)
                    Text("Radius")
                    Slider(value: $filterRadius)
                    // Add 'onChange' to watch for live changes of radius slider
                        .onChange(of: filterRadius, applyProcessing)
                        // Disable button if parameter not present in Keys
                        .disabled(!radiusPresent)
                }
                // Reduce visibility if parameter not present in Keys
                .opacity(!radiusPresent ? 0.3 : 1)
                HStack {
                    Text("Scale")
                    Slider(value: $filterScale)
                    // Add 'onChange' to watch for live changes of scale slider
                        .onChange(of: filterScale, applyProcessing)
                        // Disable button if parameter not present in Keys
                        .disabled(!scalePresent)
                }
                // Reduce visibility if parameter not present in Keys
                .opacity(!scalePresent ? 0.3 : 1)
                    .padding(.bottom)
                
                HStack {
                    // Add filter button, but extract function outside of layout
                    Button("Change Filter", action: changeFilter).disabled(selectedItem == nil)
                    
                    Spacer()
                    }
                    
                    // Share the picture
                    if let processedImage {
                        ShareLink(item: processedImage,
                            preview: SharePreview("Instafilter image",
                                image: processedImage)
                        )
                    }
                }
            }
            .padding([.horizontal, .bottom])
            .navigationTitle("Instafilter")
            // Dialog to choose filter
            .confirmationDialog(
                "Select a filter",
                isPresented: $showingFilters) {
                    Button("Crystallize") { setFilter(CIFilter.crystallize()) }
                    Button("Edges") { setFilter(CIFilter.edges()) }
                    Button("Gaussian Blur") { setFilter(CIFilter.gaussianBlur()) }
                    Button("Pixellate") { setFilter(CIFilter.pixellate()) }
                    Button("Sepia Tone") { setFilter(CIFilter.sepiaTone()) }
                    Button("Unsharp Mask") { setFilter(CIFilter.unsharpMask()) }
                    Button("Vignette") { setFilter(CIFilter.vignette()) }
                    Button("Cancel", role: .cancel) { }
                }
    }
    // Move button action here (out of layout)
    func changeFilter() {
        showingFilters = true
     }
    
    func loadImage() {
        Task {
            // Load selected image as pure 'data' and not an image type.
            guard let imageData = try await selectedItem?.loadTransferable(
                type: Data.self) else { return }
            // Convert data to image
            guard let inputImage = UIImage(data: imageData) else { return }
            
            // Convert UIImage to CIImage
            let beginImage = CIImage(image: inputImage)
            currentFilter.setValue(beginImage, forKey: kCIInputImageKey)
            // Apply the chosen filter
            applyProcessing()
        }
    }
    
    func applyProcessing() {
        // Due to compatibiltiy with this old code, we have
        // to use 'Float' instead of 'Double'
        // Once we change the @State, we lose access to .intensity
        //currentFilter.intensity = Float(filterIntensity)
        // Instead, we feed our filterIntensity var into a Key
        //currentFilter.setValue(filterIntensity, forKey: kCIInputIntensityKey)
        // This works for sepiaTone, but not for others, and it crashes
        
        // Fix:
        let inputKeys = currentFilter.inputKeys
        
        // Map filterIntensity to kCIInputIntensityKey
        if inputKeys.contains(kCIInputIntensityKey) {
            currentFilter.setValue(
                filterIntensity,
                forKey: kCIInputIntensityKey
            )
            // Set bool for presence of parameter in filter
            intensityPresent = true
        } else {
            // Set bool to report lack of parameter in Keys
            intensityPresent = false
        }
        // Map filterIntensity to kCIInputRadiusKey and multiply by 200
        if inputKeys.contains(kCIInputRadiusKey) {
            currentFilter.setValue(
                filterRadius * 200,
                forKey: kCIInputRadiusKey
            )
            // Set bool for presence of parameter in filter
            radiusPresent = true
        } else {
            // Set bool to report lack of parameter in Keys
            radiusPresent = false
        }
        // Map filterIntensity to kCIInputScaleKey and multipty by 100
        if inputKeys.contains(kCIInputScaleKey) {
            currentFilter.setValue(
                filterScale * 100,
                forKey: kCIInputScaleKey
            )
            // Set bool for presence of parameter in filter
            scalePresent = true
        } else {
            // Set bool to report lack of parameter in Keys
            scalePresent = false
        }

        
        // Capture outputImage (after filtering)
        guard let outputImage = currentFilter.outputImage else { return }
        // Convert to actual pixels
        guard let cgImage = context.createCGImage(outputImage, from: outputImage.extent)
                else { return }
        
        let uiImage = UIImage(cgImage: cgImage)
        processedImage = Image(uiImage: uiImage)
    }
    
    // Set main actor so that this runs on the main thread
    @MainActor
    func setFilter(_ filter: CIFilter) {
        currentFilter = filter
        loadImage()
        
        // Track usage to guage when to trigger Review Request
        filterCount += 1
        
        // Ask for Review
        if filterCount >= 20 {
            requestReview()
        }
    }
}
