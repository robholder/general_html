Settings & Options

" exrc for wasavi
" <ctrl><enter> to invoke

set nolaunchbell
set fullscreen

map! jk <esc>

" set a theme
"set theme=solarized
set theme=charcoal

set incsearch
set undolevels=1000
set smartcase
