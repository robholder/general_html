Parameter Types

(See note on compile error for float3x3 and float4x4.) 

kernel ParameterTypes : ImageComputationKernel<ePixelWise>
{
  Image<eRead, eAccessPoint, eEdgeClamped> src; // the input image
  Image<eWrite> dst; // the output image
  
  param:
      // this parameter is made available to the user
      bool bool_param;
      
      int int_param;
      int2 int2_param;
      int3 int3_param;
      int4 int4_param;
      
      float float_param;
      float2 float2_param;
      float3 float3_param;
      float4 float4_param;
      
      //The matrix params below fail on MacOS 15.5 when the GPU option is checked
      float3x3 float3x3_param;
      float4x4 float4x4_param;

  local:
      // This local variable is not exposed to the user.
      int int_array[2][3];
      float float_array[2][3];

  // In define(), parameters can be given labels and default values.
  void define() {

  }
  // The init() function is run once before any calls to process().
  void init() {

  }

  // The process() function runs over all pixel positions of the output image.
  void process() {

    dst() = src();
  }
};
