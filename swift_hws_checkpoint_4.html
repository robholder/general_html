Hacking With Swift - Checkpoint 4

The challenge: write a function that accepts an integer from 1 through 10,000, and returns the integer square root of that number. That sounds easy, but there are some catches:

  • You can’t use Swift’s built-in sqrt() function or similar – you need to find the square root yourself.
  • If the number is less than 1 or greater than 10,000 you should throw an “out of bounds” error.
  • You should only consider integer square roots – don’t worry about the square root of 3 being 1.732, for example.
  • If you can’t find the square root, throw a “no root” error.

As a reminder, if you have number X, the square root of X will be another number that, when multiplied by itself, gives X. So, the square root of 9 is 3, because 3x3 is 9, and the square root of 25 is 5, because 5x5 is 25.

My Attempt

let myNum = 36

enum InputError: Error {
    case small, large, noroot
}

func sqrtInt(_ x: Int) throws -> Int {
    if x < 1 {
        throw InputError.small
    }
    if x > 10_000 {
        throw InputError.large
    }
    for i in 1...100 {
        let candidate = i * i
        if candidate == x {
            //print("The square root integer of \(x) is \(i).")
            return i
        }
    }
    throw InputError.noroot
}

do {
    let result = try sqrtInt(myNum)
    print("Square root of \(myNum) is \(result).")
} catch InputError.small {
    print("Number has to be 1 or greater.")
} catch InputError.large {
    print("Number is too large. 10,000 is the largest acceptable value.")
} catch InputError.noroot {
    print("There is no integer square root for \(myNum).")
}

Attempt at streamlining: 

// User input:
let myNum = 36

// Define Errors:
enum InputError: Error {
    case numRange, noroot
}

func sqrtInt(_ x: Int) throws -> Int {
    // Check user input number is within range;
    if x < 1 || x > 10_000 {
        throw InputError.numRange
    }
    
    for i in 1...100 {
        let candidate = i * i
        if candidate == x {
            return i
        }
    }
    throw InputError.noroot
}

do {
    let result = try sqrtInt(myNum)
    print("Square root of \(myNum) is \(result).")
} catch InputError.numRange {
    print("Number has to be 1 or greater, with a limit of 10,000.")
} catch InputError.noroot {
    print("There is no integer square root for \(myNum).")
}

Other Examples

enum PlayError: Error {
	case cheating
	case noPlayers
}

func playGame(name: String, cheat: Bool = false) throws {
	if cheat {
		throw PlayError.cheating
	} else {
		print("Let's play a game of \(name)...")
	}
}

enum PizzaErrors: Error {
	case hasPineapple
}
func makePizza(type: String) throws {
	if type != "Hawaiian" {
		print("Your pizza will be ready in 10 minutes.")
	} else {
		throw PizzaErrors.hasPineapple
	}
}

enum ChargeError: Error {
    case noCable
    case noPower
}
func chargePhone(atHome: Bool) throws {
    if atHome {
        print("Phone is charging...")
    } else {
        throw ChargeError.noPower
    }
}
do {
    try chargePhone(atHome: false)
} catch ChargeError.noPower {
    print("No power socket available!")
}

enum WifiError: Error {
	case badPassword
	case noSignal
}
func connectToWifi(_ password: String) throws {
	if password == "" {
		throw WifiError.badPassword
	}
	print("You're connected.")
}

do {
    try connectToWifi("test")
} catch {
    print("You need a password.")
}

enum PrintError: Error {
	case invalidCount
}
func printPages(text: String, count: Int) throws {
	if count <= 0 {
		throw PrintError.invalidCount
	} else {
		for _ in 1...count {
			print("Printing \(text)...")
		}
	}
}

enum LoginError: Error {
	case unknownUser
}
func authenticate(username: String) throws {
	if username == "Anonymous" {
		throw LoginError.unknownUser
	}
	print("Welcome, \(username)!")
}

enum BookErrors: Error { case tooFewPages
	case tooManyPages
}
func writeBook(title: String, pages: Int) throws {
	switch pages {
	case 0...50:
		throw BookErrors.tooFewPages
	case 51...400:
		print("Perfect! I'm going to write \(title)...")
	default:
		throw BookErrors.tooManyPages
	}
}

enum CatProblems: Error {
	case notACat
	case unfriendly
}
func strokeCat(_ name: String) throws {
	if name == "Mr Bitey" {
		throw CatProblems.unfriendly
	} else if name == "Lassie" {
		throw CatProblems.notACat
	} else {
		print("You stroked \(name).")
	}
}
