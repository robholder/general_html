Install

  • In a shell (or venv): 
    python3 -m pip install bpython
  • Run bpython: 
    bpython
  • Install a config file: 
    # While in bpython (run with bpython):
<F3>
# Default path is: ~/.config/bpython/config
# $XDG_CONFIG_HOME/bpython/config
  • Expand history (in ~/.config/bpython/config): 
    # ~/.config/bpython/config
# ...

[general]
hist_length = 999_999_999_999
