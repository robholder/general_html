Vimwiki Colorscheme Fixes

" Colorscheme fix for Vimwiki (for some colorschemes)
" From Copilot: This effectively cures Vimwiki highlighting on VimwikiLinks
" and VimwikiDelimiters:
autocmd ColorScheme * highlight SpellBad guifg=NONE guibg=NONE guisp=NONE gui=NONE cterm=NONE
autocmd FileType vimwiki highlight SpellBad guifg=NONE guibg=NONE guisp=NONE gui=NONE cterm=NONE
autocmd FileType vimwiki highlight Conceal guifg=NONE guibg=NONE guisp=NONE gui=NONE cterm=NONE

Vimwiki - Set Default Colorscheme

"---------------------- Colorscheme For Vimwiki - START -----------------------"
" Set default colorscheme (already set elsewhere in this vimrc):
"let g:default_colorscheme = '<a chosen default scheme>'
" Set colorscheme for Vimwiki pages:
let g:vimwiki_colorscheme = 'nord'
"
" Change colorscheme to when switching to Vimwiki.  (Sets back to the default
" colorscheme when leaving vimwiki pages using the g:default_colorscheme
" variable set up earlier in the vimrc).
augroup VimwikiColorscheme
  " Clear any existing autocommands from this group
  autocmd!  
  " When entering a Vimwiki buffer, switch to the Vimwiki-specific colorscheme.
  autocmd FileType vimwiki exe 'colorscheme ' . g:vimwiki_colorscheme
  " When backspacing to a previous Vimwiki link/buffer, continue to set color.
  " If the buffer is NOT in the Vimwiki directory path or not a Vimwiki file,
  " revert to default.
  " NOTE: Make sure to edit the path directory: in my case '/vimwiki_dir/' matches
  " a directory on my vimwiki files path:
  autocmd BufEnter * if expand('%:p') =~# '/vimwiki_dir/' | exe 'colorscheme ' . g:vimwiki_colorscheme | else | exe 'colorscheme ' . g:default_colorscheme | endif
  " Run the SetSpellBadSettings function to set the SpellBad color settings to
  " make spelling error colors more subtle.
  autocmd FileType vimwiki call SetSpellBadSettings()
  autocmd BufLeave * if &filetype ==# 'vimwiki' | call SetSpellBadSettings() | endif
augroup END
"---------------------- Colorscheme For Vimwiki - END -------------------------"
