Python - lseq

This is a Python3 tool to list sequences of files in a directory, as a range syntax based on file stems and numbered files. Inspired by a tool written by Andrew Whitehurst. 

#!/usr/bin/env python3

import sys
import os
import argparse
from collections import defaultdict

"""
lseq: Scans a directory and returns a compact summary of frame sequences
      and dynamically highlights missing frames in red.
"""

__author__ = "My Name"
__copyright__ = "Copyright 2026, My Name"
__license__ = "MIT"
__version__ = "2.0.1"
__maintainer__ = "My Name"
__email__ = ""
__status__ = "Production"

# ANSI escape codes for coloring terminal text
COLOR_RED = "\033[91m"
COLOR_RESET = "\033[0m"

# ---------------------------------------------------------------------
# SAFETY VALVE: Prevents hanging on non-sequential numbers
# ---------------------------------------------------------------------
# No standard VFX sequence will have a gap larger than 100,000 frames.
# This prevents O(N) range allocations from locking up the system.
MAX_ALLOWED_GAP = 100000


def format_single_frame(val, pad):
    """
    Formats a single frame number, preserving its absolute padding and wrapping
    negative numbers in parentheses.
    """
    if val < 0:
        # Wrap negative numbers in parentheses, e.g., (-003)
        padded_digits = str(abs(val)).zfill(pad)
        return f"(-{padded_digits})"
    else:
        # Standard positive padding, e.g., 002
        return str(val).zfill(pad)


def makeFrameList(inputList, pad):
    """
    Takes a sorted list of integer frame numbers and aggregates them into a
    compact string, grouping consecutive runs inside square brackets.
    """
    if not inputList:
        return ""
    if len(inputList) == 1:
        return format_single_frame(inputList[0], pad)

    segments = []
    start = inputList[0]
    prev = inputList[0]

    for val in inputList[1:]:
        if val == prev + 1:
            prev = val
        else:
            if start == prev:
                segments.append(format_single_frame(start, pad))
            else:
                segments.append(f"[{format_single_frame(start, pad)}-{format_single_frame(prev, pad)}]")
            start = val
            prev = val

    # Handle final segment
    if start == prev:
        segments.append(format_single_frame(start, pad))
    else:
        segments.append(f"[{format_single_frame(start, pad)}-{format_single_frame(prev, pad)}]")

    return ",".join(segments)


def makeMissingFrameList(missingList, pad):
    """
    Groups missing frames into contiguous runs and formats them,
    enclosing ALL segments (including single frames) in square brackets,
    separated by ', '.
    """
    if not missingList:
        return ""
    if len(missingList) == 1:
        return f"[{format_single_frame(missingList[0], pad)}]"

    segments = []
    start = missingList[0]
    prev = missingList[0]

    for val in missingList[1:]:
        if val == prev + 1:
            prev = val
        else:
            if start == prev:
                segments.append(f"[{format_single_frame(start, pad)}]")
            else:
                segments.append(f"[{format_single_frame(start, pad)}-{format_single_frame(prev, pad)}]")
            start = val
            prev = val

    # Handle final segment
    if start == prev:
        segments.append(f"[{format_single_frame(start, pad)}]")
    else:
        segments.append(f"[{format_single_frame(start, pad)}-{format_single_frame(prev, pad)}]")

    return ", ".join(segments)


def stringToInt(inputString):
    """
    Safely parses a string value into a signed integer, accounting for leading negative signs.
    """
    if not inputString:
        return 0
    
    tweakVal = inputString
    if inputString.startswith("-"):
        multVal = -1
        tweakVal = inputString[1:]
    else:
        multVal = 1
        
    if not tweakVal.isdigit():
        print(f"Error: {tweakVal} is a non-numeric frame value.", file=sys.stderr)
        return 0
    else:
        return int(tweakVal) * multVal


def getFileName(inputString):
    """
    Extracts the root name stem of the sequence, cleanly stripping any trailing
    minus signs or separator dots.
    """
    base = os.path.basename(inputString)
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    idx = len(name_part)
    while idx > 0 and name_part[idx-1].isdigit():
        idx -= 1
        
    outString = name_part[:idx]
    
    # Strip the trailing minus sign if it was a negative frame
    if outString.endswith("-"):
        outString = outString[:-1]  
        
    # Strip the separator dot so we don't return "stem." as the stem name
    if outString.endswith("."):
        outString = outString[:-1]
        
    return outString


def getFrameNumber(inputString):
    """
    Extracts the frame number string from a path.
    """
    base = os.path.basename(inputString)
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    idx = len(name_part)
    while idx > 0 and (name_part[idx-1].isdigit() or name_part[idx-1] == "-"):
        if name_part[idx-1] == "-" and (idx - 1 > 0 and name_part[idx-2].isdigit()):
            break
        idx -= 1
        
    return name_part[idx:]


def getExtension(inputString):
    """
    Extracts the file extension of a path.
    """
    ext = inputString.rsplit(".", 1)
    return "." + ext[1] if len(ext) > 1 else ""


def getFramePadding(inputFrame):
    """
    Evaluates only the absolute digit padding width.
    Negative signs are ignored so that '-003' and '002' are grouped into 
    the same padding bin (width of 3).
    """
    if not inputFrame:
        return 0
    
    # Strip leading minus sign to count absolute digits
    clean_frame = inputFrame[1:] if inputFrame.startswith("-") else inputFrame
    return len(clean_frame)


# =====================================================================
# MAIN EXECUTION
# =====================================================================

def main():
    # Initialize the standard argument parser
    parser = argparse.ArgumentParser(
        description="List numbered sequences in a directory using a compact, readable syntax.",
        formatter_class=argparse.RawTextHelpFormatter
    )

    # Expose the metadata version directly to the CLI
    parser.add_argument(
        "-v", "--version",
        action="version",
        version=f"%(prog)s {__version__}"
    )

    # Define the directory argument as an optional positional argument
    parser.add_argument(
        "directory", 
        nargs="?", 
        default=".", 
        help="The directory path to scan. Defaults to the current working directory."
    )

    args = parser.parse_args()

    # Validate directory path
    testPath = args.directory
    if not os.path.exists(testPath):
        print(f"Error: Specified directory path does not exist: '{testPath}'", file=sys.stderr)
        sys.exit(1)
    if not os.path.isdir(testPath):
        print(f"Error: Specified path is a file, not a directory: '{testPath}'", file=sys.stderr)
        sys.exit(1)

    # Ensure trailing folder slash
    if not testPath.endswith(os.sep):
        testPath += os.sep

    # Locate files reliably using os.listdir
    try:
        raw_files = os.listdir(testPath)
    except OSError as e:
        print(f"Error: Unable to read directory '{testPath}': {e}", file=sys.stderr)
        sys.exit(1)

    # Filter to files with an extension (equivalent to glob's *.*)
    fileList = [os.path.join(testPath, f) for f in raw_files if "." in f and os.path.isfile(os.path.join(testPath, f))]

    if len(fileList) < 1:
        print(f"Warning: No valid sequential files found in '{testPath}'", file=sys.stderr)
        sys.exit(0)

    # Group files by signature: (name_stem, absolute_padding_width, extension)
    sequences = defaultdict(list)

    for path in fileList:
        frame_str = getFrameNumber(path)
        if frame_str and frame_str != "NG":
            stem = getFileName(path)
            pad = getFramePadding(frame_str)
            ext = getExtension(path)
            frame_val = stringToInt(frame_str)
            
            key = (stem, pad, ext)
            sequences[key].append(frame_val)

    # Accumulate output
    outputString = ""

    for (stem, pad, ext), frames in sorted(sequences.items()):
        frames.sort()
        
        # Process ranges
        formatted_frames = makeFrameList(frames, pad)
        
        # Clean, single-dot formatted line
        line = f"{stem}.{formatted_frames}{ext}\n"

        # Check for missing files if the sequence has more than one frame
        has_gaps = False
        missing_report = ""
        
        if len(frames) > 1:
            min_frame = frames[0]
            max_frame = frames[-1]
            total_gap = max_frame - min_frame
            
            # -------------------------------------------------------------
            # SAFETY CHECK: Skip analysis if sequence gap is unrealistic
            # -------------------------------------------------------------
            if total_gap <= MAX_ALLOWED_GAP:
                # Use sets to find mathematical differences
                expected_set = set(range(min_frame, max_frame + 1))
                present_set = set(frames)
                missing_set = expected_set - present_set
                
                if missing_set:
                    has_gaps = True
                    missing_list = sorted(list(missing_set))
                    missing_count = len(missing_list)
                    formatted_missing = makeMissingFrameList(missing_list, pad)
                    
                    # Format pluralization and construct the secondary info line
                    plural_suffix = "s" if missing_count > 1 else ""
                    missing_report = f"    * missing {missing_count} file{plural_suffix}: {formatted_missing}\n"

        # Output logic
        if has_gaps:
            # Color both the sequence name and its missing file report red
            outputString += f"{COLOR_RED}{line}{missing_report}{COLOR_RESET}"
        else:
            outputString += line

    # Output final report
    print(outputString, end="")


if __name__ == "__main__":
    main()

Let’s break this down step-by-step. This block of code is the "brain" of the script: its job is to look at a list of raw filenames and organize them into organized sequence groups.

To understand what is being built, let's trace this block using a real-world example.

Imagine our fileList contains these three raw file paths:

  • /path/to/render.0001.exr
  • /path/to/render.0002.exr
  • /path/to/notes.txt (not part of a sequence)

Step 1: Initialize the Container

    sequences = defaultdict(list)

Because of defaultdict(list), if we look up a sequence we haven't seen yet, Python will instantly create an empty list [] for it.

  • Current state of sequences: {} (completely empty)

Step 2: Loop Through Each File

The script starts looping through the files one by one. 

    for path in fileList:

Iteration 1: Processing /path/to/render.0001.exr

Step 2a: Extract the frame number

frame_str = getFrameNumber(path)

This calls the helper function getFrameNumber(). It strips away the path, name, and extension, leaving just the numbers at the end.

  • frame_str becomes "0001".

Step 2b: Is it a valid frame?

if frame_str and frame_str != "NG":

("NG" is a VFX industry term for "No Good" or bad frames). Since "0001" is valid, we proceed inside the if block.

Step 2c: Extract the metadata

Now, the script calls several helper functions to dissect the file path: 

stem = getFileName(path)          # Returns "render"
pad = getFramePadding(frame_str)  # Returns 4 (because "0001" is 4 digits)
ext = getExtension(path)          # Returns ".exr"
frame_val = stringToInt(frame_str) # Converts "0001" to the integer 1

Step 2d: Build the "Unique Key"

This is the secret sauce of the script. To group sequences together, it creates a Python "tuple" (a fixed list) consisting of the (name, padding, extension): 

key = (stem, pad, ext)
# key becomes: ("render", 4, ".exr")

Step 2e: Store the frame number

sequences[key].append(frame_val)

Python looks for ("render", 4, ".exr") in our dictionary. It isn't there, so defaultdict automatically creates an empty list and appends our integer 1 to it.

Current state of sequences:

{
    ("render", 4, ".exr"): [1]
}

Iteration 2: Processing /path/to/render.0002.exr

The loop moves to the second file.

  • frame_str becomes "0002".
  • It is valid, so we dissect it:
    • stem = "render"
    • pad = 4
    • ext = ".exr"
    • frame_val = 2
  • The unique key is built: key = ("render", 4, ".exr")
  • sequences[key].append(frame_val) is called.
  • What happens here: The dictionary already has this key! Python grabs the existing list [1] and appends 2 to it.
  • Current state of sequences: 
{
    ("render", 4, ".exr"): [1, 2]
}

Iteration 3: Processing /path/to/notes.txt

The loop moves to the final file.

  • frame_str = getFrameNumber(path) is run.
  • Because notes.txt has no trailing numbers, frame_str returns "" (an empty string).
  • The check if frame_str: fails.
  • What happens here: The script completely skips this file and does nothing.

The Final Result

Once the loop finishes, the sequences dictionary has completely organized your files. It now looks like this: 

{
    ("render", 4, ".exr"): [1, 2]
}

This data structure is perfect for the next stage of the script, because it has successfully grouped all related frames into a single list, safely separated by their name, padding, and file extension!

Defaultdict

In Python, defaultdict is a highly convenient subclass of the standard dictionary (dict) found in the built-in collections module.

Its sole purpose is to prevent KeyError exceptions by automatically creating a default value whenever you try to access or modify a key that doesn't exist yet.

The Problem with Standard Dictionaries

With a standard Python dictionary, if you try to append something to a list inside a key that hasn't been created yet, Python will crash: 

# This will CRASH with a KeyError:
sequences = {}
key = ("render_layer", 4, ".exr")

sequences[key].append(101)  # ❌ KeyError! The key doesn't exist yet.

To make this work with a standard dictionary, you have to manually check for the key every single time: 

# The manual way:
if key not in sequences:
    sequences[key] = []  # Initialize it first
sequences[key].append(101)

How defaultdict Solves This

When you initialize a defaultdict, you pass it a "default factory" - a callable object (like list, int, or set) that tells the dictionary what to put there if a key is missing.

In a script: 

sequences = defaultdict(list)

By passing list as the factory, you are telling Python: "If I ask for a key that isn't there, instantly create it and set its value to an empty list []."

This allows you to safely write this single line: 

sequences[key].append(frame_val)

What happens behind the scenes:

  • First time encountering a sequence (e.g., ("shot_01", 4, ".exr")):
    • defaultdict notices this key is missing.
    • It quietly runs list() to generate an empty list [].
    • It inserts the key: {("shot_01", 4, ".exr"): []}.
    • It immediately appends your frame number (e.g., 101) to that new list.
  • Second time encountering the same sequence:
  • defaultdict sees the key already exists.
  • It skips initialization and simply appends the next frame number (e.g., 102) to the existing list.

Other Common Uses for defaultdict

You can use different factories depending on what you are trying to group or count:

  • defaultdict(int): Great for counting things. Missing keys start at 0. 
counts = defaultdict(int)
counts["apple"] += 1  # Automatically starts at 0, becomes 1
  • defaultdict(set): Great for gathering unique, non-duplicate items. Missing keys start as set(). 
    unique_exts = defaultdict(set)
unique_exts["render_passes"].add(".exr")
