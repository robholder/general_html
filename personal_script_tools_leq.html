Python - lseq

This is a modern Python3 reinterpretation of an excellent tool originally made by Andrew Whitehurst. This has been re-written in Python3 (original was in Python2), and a few logic tweaks. 

#!/usr/bin/env python3

import sys
import os
import glob
from collections import defaultdict

# lseq IS A UTILITY THAT SCANS A DIRECTORY AND RETURNS FORMATTED LISTS OF
# SEQUENCES OF FRAMES MAKING IT EASIER TO SEE WHICH FRAMES ARE MISSING,
# WHAT THE FRAME RANGES ARE AND SO ON. IT RETURNS THE LIST IN A READABLE
# FORMATTING

# negPad handles whether we regard -#### as four or five frame padding
negPad = 0


def makeFrameList(inputList, pad):
    outputString = ""
    if len(inputList) > 1:
        lastTestedNumber = inputList[0]
        minRun = lastTestedNumber  # Low number in a run of consecutive frames
        
        # Copy list to avoid mutating the original
        working_list = list(inputList)
        del working_list[0]
        
        # Append a sentinel value so the loop processes the last run
        working_list.append(working_list[-1] + 10)
        maxRun = minRun            # High value in a consecutive run of frames
        
        for x in working_list:
            if x == (lastTestedNumber + 1):
                maxRun = x
                lastTestedNumber = x
            else:
                if maxRun == minRun:  # Single frame all on its own
                    if maxRun < 0:
                        outputString += str(maxRun).zfill(pad + 1) + ","
                    else:
                        outputString += str(maxRun).zfill(pad) + ","
                    minRun = x
                    maxRun = x                
                elif maxRun != minRun and minRun < maxRun:  # Legitimate run of frames
                    outputString += f"{minRun}-{maxRun},"
                    minRun = x    
                    maxRun = x
                lastTestedNumber = x
        outputString = outputString[:-1]  # Strip trailing comma
        
    elif len(inputList) == 1:  # Single-frame sequence
        if inputList[0] < 0:
            outputString = str(inputList[0]).zfill(pad + negPad)
        else:
            outputString = str(inputList[0]).zfill(pad)

    return outputString


def errorMessage():
    print("\n\tUSAGE : lseq [OPTION] [DIRECTORY]\n")
    print("\tlseq will list numbered sequences in an easy to read format")
    print("\tthat will show ranges of existing files. This enables the user")
    print("\tto easily see where files are missing and to see multiple image")
    print("\tsequences contained within the same directory. The list is")
    print("\tin a compact sequence format, i.e. file.1-7,0010,0015,18-20#.ext\n")
    print("\tlseq will default to treating negative frames like abc.-001.ext as")
    print("\ta four frame padded number. If you wish to treat such a frame as ")
    print("\tthree frame padding add the -oldPad flag option.\n")
    print("\tAn optional valid directory path may be specified to run the")
    print("\tcommand on.\n\n")
    sys.exit(1)


def stringToInt(inputString):
    if not inputString:
        return 0
    
    tweakVal = inputString
    if inputString.startswith("-"):
        multVal = -1
        tweakVal = inputString[1:]
    else:
        multVal = 1
        
    if not tweakVal.isdigit():
        print(f"{tweakVal} is a non-numeric frame value.")
        return 0
    else:
        return int(tweakVal) * multVal


def getFileName(inputString):
    base = os.path.basename(inputString)
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    # Work backward to find where numbers stop
    idx = len(name_part)
    while idx > 0 and name_part[idx-1].isdigit():
        idx -= 1
        
    outString = name_part[:idx]
    if outString.endswith("-"):
        outString = outString[:-1]  # Remove minus sign from negative frame stems
        
    return outString


def getFrameNumber(inputString):
    base = os.path.basename(inputString)
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    idx = len(name_part)
    while idx > 0 and (name_part[idx-1].isdigit() or name_part[idx-1] == "-"):
        # Make sure we don't treat a middle dash as a negative sign
        if name_part[idx-1] == "-" and (idx - 1 > 0 and name_part[idx-2].isdigit()):
            break
        idx -= 1
        
    return name_part[idx:]


def getExtension(inputString):
    ext = inputString.rsplit(".", 1)
    return "." + ext[1] if len(ext) > 1 else ""


def getFramePadding(inputFrame):
    if not inputFrame:
        return 0
    # Corrected padding calculation logic
    if len(inputFrame) > 1 and inputFrame.startswith("-") and inputFrame[1] == "0":
        return len(inputFrame) - negPad
    if len(inputFrame) > 1 and inputFrame.startswith("0"):
        return len(inputFrame)
    return len(inputFrame)


# ==========================================
# MAIN EXECUTION
# ==========================================

if len(sys.argv) > 3:
    errorMessage()

if len(sys.argv) >= 2:
    if sys.argv[1] in ["-h", "-help", "-H", "--help"]:
        errorMessage()
    if sys.argv[1] == "-oldPad":
        negPad = 1

# Build the directory path to test
testPath = ""
if len(sys.argv) >= 2 and sys.argv[-1] != "-oldPad":
    target = sys.argv[-1]
    if os.path.exists(target):
        testPath = target
        # Ensure target ends with a slash so glob works cleanly
        if not testPath.endswith(os.sep):
            testPath += os.sep
    else:
        errorMessage()

# Gather files matching pattern
fileList = glob.glob(testPath + "*.*")

if len(fileList) < 1:
    print("\n\tNo valid files found.\n")
    sys.exit(1)

# Group matching files by their unique (stem, pad, extension) signature
sequences = defaultdict(list)

for path in fileList:
    frame_str = getFrameNumber(path)
    if frame_str and frame_str != "NG":
        stem = getFileName(path)
        pad = getFramePadding(frame_str)
        ext = getExtension(path)
        frame_val = stringToInt(frame_str)
        
        # Unique identifier group
        key = (stem, pad, ext)
        sequences[key].append(frame_val)

# Format and generate the final syntax
outputString = "\n"

for (stem, pad, ext), frames in sorted(sequences.items()):
    frames.sort()
    
    # Process sequence list to ranges
    formatted_frames = makeFrameList(frames, pad)
    
    if len(frames) > 1:
        outputString += f"\t{stem}{formatted_frames}#{ext}\n"
    else:
        outputString += f"\t{stem}{formatted_frames}{ext}\n"

print(outputString)

Comprehensive breakdown of how the lseq script operates.

1. Overview and Approach

In industries like visual effects (VFX), animation, and game development, image sequences (e.g., car_render.0001.exr through car_render.0100.exr) are the standard way to handle video. When rendering thousands of frames, it is incredibly easy for a few frames to fail or go missing.

The purpose of lseq is to scan a directory, detect these sequences, and compress them into a highly readable syntax.

Instead of showing you a massive, unreadable wall of 500 files, it groups them into elegant patterns:

  • Consecutive sequences: render.0001-0010#.exr
  • Single frame gaps: render.0001-0010,0012,0015-0020#.exr
  • Isolated files: image.png (no sequence formatting applied)

The Three-Step Approach

[Raw Files] ────> [Step 1: Deconstruct] ────> [Step 2: Group & Sort] ────> [Step 3: Format Ranges]
- file.0001.exr   Stem: "file"                Key: ("file", 4, ".exr")      "file.0001-0003,0005#.exr"
- file.0002.exr   Pad:  4                     Frames: [1, 2, 3, 5]
- file.0003.exr   Ext:  ".exr"
- file.0005.exr   Frame: 1, 2, 3, 5
  • Deconstruct: The script looks at every file name and extracts its core structural properties: the name stem, the frame number, the padding (amount of leading zeros), and the file extension.
  • Group and Sort: It groups files sharing the exact same name stem, padding, and extension. Then, it sorts the frame numbers numerically.
  • Format Ranges: It loops through the sorted frame numbers to identify contiguous "runs" (consecutive numbers) and formats them into readable ranges (e.g., 0001-0010) while preserving their original padding.

2. Step-by-Step Code Breakdown

Step 1: The Helper Functions (The Parsers)

These functions slice and dice individual file paths into metadata components.

getFileName(inputString)

  • What it does: Extracts the pure name of the file (the "stem") before the numbers start.
  • How it works: It strips the extension, then walks backward (right-to-left) through the characters. Once it hits a non-digit character, it knows the frame number has ended. It cuts the string there. If a minus sign remains at the end (e.g., file.-001.exr), it removes it so that negative frames share the same stem as positive ones.

getFrameNumber(inputString)

  • What it does: Extracts the literal digits of the frame number (including a minus sign if it is a negative frame).
  • How it works: Similar to getFileName, it walks backward from the end of the filename. It grabs all adjacent digits and a single preceding negative sign, returning it as a string (e.g., "0042" or "-005").

getFramePadding(inputFrame)

  • What it does: Determines the character length of the frame number to preserve leading zeros.
  • How it works: If the string starts with 0 (like 0042), its length is 4. It also contains logic for handling negative padding (defined by the -oldPad CLI flag) so that negative frames (like -001) are calculated with the correct column width.

stringToInt(inputString)

  • What it does: Converts a string frame number like "-0023" into a true mathematical integer (-23).
  • How it works: It strips out any negative signs to check if the remaining characters are digits, then performs the conversion and multiplies by -1 if it was negative.

Step 2: Main Routine & File Grouping (The Orchestrator)

This is the main body of the script that executes when you run it. 

# Gather files matching pattern
fileList = glob.glob(testPath + "*.*")
  • Directory Resolution: The script checks if you supplied a directory argument. If not, it defaults to the current directory. It uses glob.glob to find all files containing at least one dot (*.*). 
sequences = defaultdict(list)
for path in fileList:
    ...
    key = (stem, pad, ext)
    sequences[key].append(frame_val)
  • Grouping using a Dictionary (defaultdict): This is a highly efficient Python 3 pattern. It creates a dictionary where the keys are a tuple of unique identifiers: (stem, padding, extension). As it loops through the files, it appends each frame integer to the list mapped to that specific key.
  • Example: Files image.0001.png and image.0002.png both map to the key ("image", 4, ".png"). Their integers [1, 2] are grouped together in a list.

Step 3: Range Assembly (The Engine)

Once all files are grouped, the script loops through each unique sequence key, sorts the list of frame numbers, and hands them off to the core formatting function.

makeFrameList(inputList, pad)

This function is the heart of the script's logic. It turns [1, 2, 3, 5, 10, 11, 12] into "0001-0003,0005,0010-0012".

  • The Sentinel Value (Preventing Edge-Case Drops): The script appends a dummy value to the end of the list: 
working_list.append(working_list[-1] + 10)

Because the algorithm determines whether to write a range only when the consecutive chain breaks, a sequence that ends perfectly would never trigger a write. This "sentinel" value guarantees a break at the very end, forcing the final real sequence range to compile.

  • The Consecutive Check Loop:
    • It iterates through the sorted numbers. If the current number x is exactly lastTestedNumber + 1, it updates maxRun (extending the current chain).
  • The Breakpoint Check:
    • When x is not consecutive:
      • If maxRun == minRun: Only one file was in this block (a gap or single frame). It pads the single frame (e.g., 0005) and appends it with a comma.
      • If minRun < maxRun: A consecutive run occurred. It writes the range (e.g., 1-3) and appends it.
      • It then resets the markers (minRun and maxRun) to the current number x to start tracking the next chain.

Finally, the script cleans up any trailing commas, formats single-file entries cleanly, and prints the completed styled sequence string directly to your terminal.

#!/usr/bin/env python3

import sys
import os
import glob
from collections import defaultdict

# lseq IS A UTILITY THAT SCANS A DIRECTORY AND RETURNS FORMATTED LISTS OF
# SEQUENCES OF FRAMES MAKING IT EASIER TO SEE WHICH FRAMES ARE MISSING,
# WHAT THE FRAME RANGES ARE AND SO ON. IT RETURNS THE LIST IN A READABLE
# FORMATTING

# This global toggle manages how negative frame numbers are padded.
# If negPad is 0: -0001 is evaluated as a 5-character string (5-digit padding).
# If negPad is 1: -0001 is evaluated as 4-digit padding (ignoring the '-' sign).
negPad = 0


def makeFrameList(inputList, pad):
    """
    Takes a sorted list of integer frame numbers and aggregates them into a
    compact, comma-separated string of single frames and ranges (e.g., '0001-0005,0008,0010-0012').
    """
    outputString = ""
    
    # We can only evaluate sequences/gaps if we have more than one frame to compare.
    if len(inputList) > 1:
        # Establish our starting points for consecutive frame tracking
        lastTestedNumber = inputList[0]
        minRun = lastTestedNumber  # Marks the lower boundary of the current sequence run
        
        # Clone the list to prevent mutating the original data structure outside the function
        working_list = list(inputList)
        
        # Remove the first index since we've already cached it in 'lastTestedNumber' and 'minRun'
        del working_list[0]
        
        # SENTINEL VALUE: Append a dummy value to force a non-consecutive break at the end.
        # Without this, the final run or frame of the list would never trigger the logic branches
        # below, leaving the end of your sequence completely truncated.
        working_list.append(working_list[-1] + 10)
        maxRun = minRun  # Marks the upper boundary of the current sequence run
        
        # Evaluate each frame number to see if the consecutive chain continues
        for x in working_list:
            # If the current frame is exactly 1 unit larger, the consecutive chain is unbroken
            if x == (lastTestedNumber + 1):
                maxRun = x
                lastTestedNumber = x
            # The chain is broken. We must now package the run we just completed.
            else:
                # SCENARIO A: The run consists of a single isolated frame
                if maxRun == minRun:
                    # Account for the extra character required by the negative sign when padding
                    if maxRun < 0:
                        outputString += str(maxRun).zfill(pad + 1) + ","
                    else:
                        outputString += str(maxRun).zfill(pad) + ","
                    # Reset the boundaries to track the next potential sequence starting at current 'x'
                    minRun = x
                    maxRun = x                
                # SCENARIO B: A legitimate consecutive block of frames was captured
                elif maxRun != minRun and minRun < maxRun:
                    outputString += f"{minRun}-{maxRun},"
                    # Reset boundaries to current 'x'
                    minRun = x    
                    maxRun = x
                
                # Keep tracking consecutive values relative to this newly started block
                lastTestedNumber = x
                
        # Slice off the trailing comma leftover from the loop's formatting
        outputString = outputString[:-1]  
        
    # If the sequence literally contains just one solitary frame, skip sequence grouping
    elif len(inputList) == 1:  
        if inputList[0] < 0:
            outputString = str(inputList[0]).zfill(pad + negPad)
        else:
            outputString = str(inputList[0]).zfill(pad)

    return outputString


def errorMessage():
    """
    Standard usage output block printed to stdout before terminating execution on error or help request.
    """
    print("\n\tUSAGE : lseq [OPTION] [DIRECTORY]\n")
    print("\tlseq will list numbered sequences in an easy to read format")
    print("\tthat will show ranges of existing files. This enables the user")
    print("\tto easily see where files are missing and to see multiple image")
    print("\tsequences contained within the same directory. The list is")
    print("\tin a compact sequence format, i.e. file.1-7,0010,0015,18-20#.ext\n")
    print("\tlseq will default to treating negative frames like abc.-001.ext as")
    print("\ta four frame padded number. If you wish to treat such a frame as ")
    print("\tthree frame padding add the -oldPad flag option.\n")
    print("\tAn optional valid directory path may be specified to run the")
    print("\tcommand on.\n\n")
    sys.exit(1)


def stringToInt(inputString):
    """
    Safely parses a string value into a signed integer, accounting for leading negative signs.
    """
    if not inputString:
        return 0
    
    tweakVal = inputString
    # Detect negative frame strings and isolate the digits to verify numerical validity
    if inputString.startswith("-"):
        multVal = -1
        tweakVal = inputString[1:]
    else:
        multVal = 1
        
    # Fallback guard if the string parsing yielded structural non-digits
    if not tweakVal.isdigit():
        print(f"{tweakVal} is a non-numeric frame value.")
        return 0
    else:
        return int(tweakVal) * multVal


def getFileName(inputString):
    """
    Extracts the root name stem of the sequence, cutting off extension and trailing frame numbers.
    Example: '/path/to/my_render.-0045.exr' -> 'my_render'
    """
    # Isolate filename from system folder path
    base = os.path.basename(inputString)
    # Split only on the last dot to cleanly separate extension
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    # Walk right-to-left backward down the filename string to determine where digits stop
    idx = len(name_part)
    while idx > 0 and name_part[idx-1].isdigit():
        idx -= 1
        
    # Extract the name up to the point where digits start
    outString = name_part[:idx]
    
    # If the frame was negative (e.g., 'render.-010'), strip the lingering negative sign
    if outString.endswith("-"):
        outString = outString[:-1]  
        
    return outString


def getFrameNumber(inputString):
    """
    Extracts the frame number string from a path.
    Example: '/path/to/my_render.-0045.exr' -> '-0045'
    """
    base = os.path.basename(inputString)
    noExt = base.rsplit(".", 1)
    name_part = noExt[0]
    
    # Walk backward right-to-left, checking characters that are numeric OR negative signs
    idx = len(name_part)
    while idx > 0 and (name_part[idx-1].isdigit() or name_part[idx-1] == "-"):
        # Prevent false negative matching (e.g. 'my-file-120.exr' should not match '-120' as negative)
        if name_part[idx-1] == "-" and (idx - 1 > 0 and name_part[idx-2].isdigit()):
            break
        idx -= 1
        
    return name_part[idx:]


def getExtension(inputString):
    """
    Extracts the file extension of a path.
    Example: 'image.0001.exr' -> '.exr'
    """
    ext = inputString.rsplit(".", 1)
    return "." + ext[1] if len(ext) > 1 else ""


def getFramePadding(inputFrame):
    """
    Evaluates the literal string length of a frame to determine its padding width.
    """
    if not inputFrame:
        return 0
        
    # Negative frame padding calculation adjusted by global `negPad` modifier
    if len(inputFrame) > 1 and inputFrame.startswith("-") and inputFrame[1] == "0":
        return len(inputFrame) - negPad
        
    # Standard positive frame padding (e.g., '0020' yields 4)
    if len(inputFrame) > 1 and inputFrame.startswith("0"):
        return len(inputFrame)
        
    return len(inputFrame)


# =====================================================================
# MAIN EXECUTION
# =====================================================================

# Quick structural CLI validation
if len(sys.argv) > 3:
    errorMessage()

# Check for help flags
if len(sys.argv) >= 2:
    if sys.argv[1] in ["-h", "-help", "-H", "--help"]:
        errorMessage()
    # Check if the user specified the old legacy negative padding scheme
    if sys.argv[1] == "-oldPad":
        negPad = 1

# Extract target path directory (defaulting to Current Working Directory if omitted)
testPath = ""
if len(sys.argv) >= 2 and sys.argv[-1] != "-oldPad":
    target = sys.argv[-1]
    if os.path.exists(target):
        testPath = target
        # Crucial slash check: ensuring paths have trailing slashes prevents glob from fail-matching
        if not testPath.endswith(os.sep):
            testPath += os.sep
    else:
        errorMessage()

# Locate matching files using simple globbing
fileList = glob.glob(testPath + "*.*")

if len(fileList) < 1:
    print("\n\tNo valid files found.\n")
    sys.exit(1)

# A dictionary that automatically initializes list values for new keys.
# This groups all files by their signature: (name_stem, padding_width, extension).
sequences = defaultdict(list)

for path in fileList:
    frame_str = getFrameNumber(path)
    # Ignore non-sequential files (files lacking structural sequence numbering)
    if frame_str and frame_str != "NG":
        stem = getFileName(path)
        pad = getFramePadding(frame_str)
        ext = getExtension(path)
        frame_val = stringToInt(frame_str)
        
        # The unique signature key that groups relative files together
        key = (stem, pad, ext)
        sequences[key].append(frame_val)

# Accumulate the finalized string data to print out
outputString = "\n"

# Iterate through sorted sequences ensuring deterministic command line output
for (stem, pad, ext), frames in sorted(sequences.items()):
    # Sort frame numbers in ascending order so sequential gaps and runs can be analyzed
    frames.sort()
    
    # Process the sorted list of frame values into a concise, grouped range list string
    formatted_frames = makeFrameList(frames, pad)
    
    # Multi-file sequences get the sequence syntax '#' marker appended
    if len(frames) > 1:
        outputString += f"\t{stem}{formatted_frames}#{ext}\n"
    # Individual files do not receive range formatting markers
    else:
        outputString += f"\t{stem}{formatted_frames}{ext}\n"

# Output the finished report to the terminal
print(outputString)
