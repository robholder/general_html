🌟 What --migrate does in this script

This mode:

✔ Scans your $HOME for dotfiles

Only files and directories beginning with . (excluding things like .cache, .local, .Trash, .zsh_sessions, etc.)

✔ Maps each file to a package

Example: 

~/.zshrc → dotfiles/zshrc/dot-zshrc
~/.vimrc → dotfiles/vim/dot-vimrc
~/.config/wezterm → dotfiles/wezterm/dot-config/wezterm

✔ Creates missing package directories

If dotfiles/vim/ doesn’t exist, it creates it.

✔ Shows you a migration plan

Nothing happens until you confirm.

✔ Backs up every file before adopting

Example: 

~/.zshrc → ~/.zshrc.backup.20260225-0041

✔ Runs stow --adopt package‑by‑package

This moves files into your repo and replaces them with symlinks.

✔ Does NOT commit anything

You stay in control.

Here it is — clean, safe, cross‑platform, and fully featured.

#!/usr/bin/env bash
set -e

DOTFILES_DIR="$HOME/dotfiles"
TARGET="$HOME"

# Flags
FORCE=0
UNINSTALL=0
DRYRUN=0
ADOPT=0
VERBOSE=0
LIST=0
ONLY=""
MIGRATE=0

# Colors
RED="\033[0;31m"
GREEN="\033[0;32m"
YELLOW="\033[1;33m"
BLUE="\033[0;34m"
RESET="\033[0m"

info()    { echo -e "${BLUE}INFO:${RESET} $1"; }
warn()    { echo -e "${YELLOW}WARN:${RESET} $1"; }
error()   { echo -e "${RED}ERROR:${RESET} $1"; }
success() { echo -e "${GREEN}OK:${RESET} $1"; }

# Cross-platform package detection
detect_packages() {
  for dir in "$DOTFILES_DIR"/*; do
    [ -d "$dir" ] && basename "$dir"
  done
}

# Detect dotfiles in $HOME (excluding junk)
detect_home_dotfiles() {
  find "$HOME" -maxdepth 1 -mindepth 1 -name ".*" | while read -r file; do
    base=$(basename "$file")

    case "$base" in
      .|..|.git|.cache|.local|.Trash|.zsh_sessions|.DS_Store)
        continue
        ;;
    esac

    echo "$file"
  done
}

check_conflicts() {
  local pkg="$1"
  stow --dotfiles -nvt "$TARGET" "$pkg" 2>&1 | grep "existing target" || true
}

backup_file() {
  local file="$1"
  local backup="$file.backup.$(date +%Y%m%d%H%M%S)"
  warn "Backing up $file → $backup"
  mv "$file" "$backup"
}

ensure_directories() {
  info "Ensuring required directories exist"
  mkdir -p "$HOME/.config"
  mkdir -p "$HOME/.local/share"
}

stow_package() {
  local pkg="$1"

  info "Processing package: $pkg"

  local conflicts
  conflicts=$(check_conflicts "$pkg")

  if [[ -n "$conflicts" ]]; then
    warn "Conflicts detected:"
    echo "$conflicts"

    if [[ "$FORCE" == "1" ]]; then
      warn "Force mode enabled — backing up conflicting files"
      echo "$conflicts" | awk '{print $NF}' | while read -r file; do
        backup_file "$TARGET/$file"
      done
    else
      error "Use --force to automatically back up and continue"
      return
    fi
  fi

  if [[ "$DRYRUN" == "1" ]]; then
    info "[DRY RUN] Would stow $pkg"
    stow --dotfiles -nvt "$TARGET" "$pkg"
    return
  fi

  if [[ "$ADOPT" == "1" ]]; then
    info "Adopting existing files for $pkg"
    stow --dotfiles --adopt -vt "$TARGET" "$pkg"
  else
    stow --dotfiles -vt "$TARGET" "$pkg"
  fi

  success "Stowed $pkg"
}

unstow_package() {
  local pkg="$1"
  info "Unstowing $pkg"
  stow --dotfiles -Dt "$TARGET" "$pkg"
  success "Unstowed $pkg"
}

migrate_home() {
  info "Scanning home directory for dotfiles…"

  local files
  files=$(detect_home_dotfiles)

  if [[ -z "$files" ]]; then
    warn "No dotfiles found to migrate"
    exit 0
  fi

  echo ""
  info "Migration plan:"
  echo "$files" | while read -r file; do
    base=$(basename "$file")     # ".vim"
    clean="${base#.}"            # "vim"
    pkg="${clean%%.*}"           # "vim"

    echo "  $file → $DOTFILES_DIR/$pkg/dot-$clean"
  done

  echo ""
  read -p "Proceed with migration? (y/N): " ans
  [[ "$ans" != "y" ]] && exit 0

  echo "$files" | while read -r file; do
    base=$(basename "$file")
    clean="${base#.}"
    pkg="${clean%%.*}"

    mkdir -p "$DOTFILES_DIR/$pkg"

    target="$DOTFILES_DIR/$pkg/dot-$clean"

    if [[ -e "$target" ]]; then
      warn "Target already exists: $target"
      continue
    fi

    backup_file "$file"
    mv "$file" "$target"
    ln -s "$target" "$file"

    success "Migrated $file → $target"
  done

  success "Migration complete"
}

main() {
  cd "$DOTFILES_DIR"

  if [[ "$MIGRATE" == "1" ]]; then
    migrate_home
    exit 0
  fi

  local packages
  packages=$(detect_packages)

  if [[ "$LIST" == "1" ]]; then
    info "Detected packages:"
    echo "$packages"
    exit 0
  fi

  ensure_directories

  if [[ -n "$ONLY" ]]; then
    stow_package "$ONLY"
    exit 0
  fi

  for pkg in $packages; do
    if [[ "$UNINSTALL" == "1" ]]; then
      unstow_package "$pkg"
    else
      stow_package "$pkg"
    fi
  done

  success "All operations complete"
}

# Parse flags
while [[ $# -gt 0 ]]; do
  case "$1" in
    --force) FORCE=1 ;;
    --uninstall) UNINSTALL=1 ;;
    --dry-run) DRYRUN=1 ;;
    --adopt) ADOPT=1 ;;
    --verbose) VERBOSE=1 ;;
    --list) LIST=1 ;;
    --only) ONLY="$2"; shift ;;
    --migrate) MIGRATE=1 ;;
  esac
  shift
done

main

migrate_home() {
  info "Scanning home directory for dotfiles…"

  mapfile -t files < <(detect_home_dotfiles)

  if [[ ${#files[@]} -eq 0 ]]; then
    warn "No dotfiles found to migrate"
    exit 0
  fi

  echo ""
  info "Migration plan:"
  i=1
  for file in "${files[@]}"; do
    base=$(basename "$file")
    clean="${base#.}"
    pkg="${clean%%.*}"
    echo "  [$i] $file → $DOTFILES_DIR/$pkg/dot-$clean"
    ((i++))
  done

  echo ""
  read -p "Select items to migrate (e.g. '1 2 5', 'all', or 'none'): " selection

  if [[ "$selection" == "none" ]]; then
    warn "No items selected. Aborting migration."
    exit 0
  fi

  selected_files=()

  if [[ "$selection" == "all" ]]; then
    selected_files=("${files[@]}")
  else
    for index in $selection; do
      if [[ "$index" =~ ^[0-9]+$ ]] && (( index >= 1 && index <= ${#files[@]} )); then
        selected_files+=("${files[$((index-1))]}")
      else
        warn "Invalid selection: $index"
      fi
    done
  fi

  echo ""
  info "Migrating selected items…"

  for file in "${selected_files[@]}"; do
    base=$(basename "$file")
    clean="${base#.}"
    pkg="${clean%%.*}"

    mkdir -p "$DOTFILES_DIR/$pkg"

    target="$DOTFILES_DIR/$pkg/dot-$clean"

    if [[ -e "$target" ]]; then
      warn "Target already exists: $target"
      continue
    fi

    backup_file "$file"
    mv "$file" "$target"
    ln -s "$target" "$file"

    success "Migrated $file → $target"
  done

  success "Migration complete"
}

🌟 The Design: Dry‑run by Default

Right now, your script behaves like this:

  • Normal mode → performs actions
  • --dry-run → preview only

You want:

  • Normal mode → preview only
  • --apply (or --run, or --execute) → perform actions

This is a much safer default.

  • ✔ Dry‑run becomes the default
  • ✔ --apply becomes the “do it for real” flag
  • ✔ --force still controls conflict backups
  • ✔ --migrate still shows a plan and asks for selections

This is exactly how tools like terraform and ansible behave — preview first, apply second.

🌟 Naming the new flag

You have options:

  • --apply (my recommendation — intuitive and common)
  • --run
  • --execute
  • --commit (but this sounds like Git)
  • --do-it (fun but not serious)

Let’s go with: 

--apply

It’s clear, safe, and conventional.

After

#!/usr/bin/env bash
set -e

# Dotfiles directory = current working directory
DOTFILES_DIR="$(pwd)"
TARGET="$HOME"

# Safety check: never run from $HOME
if [[ "$DOTFILES_DIR" == "$HOME" ]]; then
  echo "ERROR: Refusing to run from \$HOME. Create a dotfiles directory first."
  exit 1
fi

# Flags
APPLY=0
FORCE=0
UNINSTALL=0
ADOPT=0
VERBOSE=0
LIST=0
ONLY=""
MIGRATE=0

# Colors
RED="\033[0;31m"
GREEN="\033[0;32m"
YELLOW="\033[1;33m"
BLUE="\033[0;34m"
RESET="\033[0m"

info()    { echo -e "${BLUE}INFO:${RESET} $1"; }
warn()    { echo -e "${YELLOW}WARN:${RESET} $1"; }
error()   { echo -e "${RED}ERROR:${RESET} $1"; }
success() { echo -e "${GREEN}OK:${RESET} $1"; }

# Load .migrateignore patterns
load_migrateignore() {
  IGNORE_PATTERNS=()
  if [[ -f "$DOTFILES_DIR/.migrateignore" ]]; then
    while IFS= read -r line; do
      [[ -z "$line" || "$line" =~ ^# ]] && continue
      IGNORE_PATTERNS+=("$line")
    done < "$DOTFILES_DIR/.migrateignore"
  fi
}

# Check if a file matches ignore patterns
is_ignored() {
  local file="$1"
  for pattern in "${IGNORE_PATTERNS[@]}"; do
    if [[ "$file" == $HOME/$pattern || "$file" == $pattern || "$file" == *$pattern* ]]; then
      return 0
    fi
  done
  return 1
}

# Detect packages
detect_packages() {
  for dir in "$DOTFILES_DIR"/*; do
    [ -d "$dir" ] && basename "$dir"
  done
}

# Detect dotfiles in $HOME
detect_home_dotfiles() {
  find "$HOME" -maxdepth 1 -mindepth 1 -name ".*" | while read -r file; do
    base=$(basename "$file")

    # Skip obvious junk
    case "$base" in
      .|..|.git|.cache|.local|.Trash|.zsh_sessions|.DS_Store)
        continue
        ;;
    esac

    # Skip ignored files
    if is_ignored "$file"; then
      continue
    fi

    echo "$file"
  done
}

backup_file() {
  local file="$1"
  local backup="$file.backup.$(date +%Y%m%d%H%M%S)"
  warn "Backing up $file → $backup"
  mv "$file" "$backup"
}

check_conflicts() {
  local pkg="$1"
  stow --dotfiles -nvt "$TARGET" "$pkg" 2>&1 | grep "existing target" || true
}

ensure_directories() {
  mkdir -p "$HOME/.config"
  mkdir -p "$HOME/.local/share"
}

stow_package() {
  local pkg="$1"

  info "Processing package: $pkg"

  local conflicts
  conflicts=$(check_conflicts "$pkg")

  if [[ -n "$conflicts" ]]; then
    warn "Conflicts detected:"
    echo "$conflicts"

    if [[ "$FORCE" == "1" ]]; then
      warn "Force mode enabled — backing up conflicting files"
      echo "$conflicts" | awk '{print $NF}' | while read -r file; do
        backup_file "$TARGET/$file"
      done
    else
      error "Use --force to automatically back up and continue"
      return
    fi
  fi

  if [[ "$APPLY" != "1" ]]; then
    info "[DRY RUN] Would stow $pkg"
    stow --dotfiles -nvt "$TARGET" "$pkg"
    return
  fi

  if [[ "$ADOPT" == "1" ]]; then
    stow --dotfiles --adopt -vt "$TARGET" "$pkg"
  else
    stow --dotfiles -vt "$TARGET" "$pkg"
  fi

  success "Stowed $pkg"
}

unstow_package() {
  local pkg="$1"

  if [[ "$APPLY" != "1" ]]; then
    info "[DRY RUN] Would unstow $pkg"
    stow --dotfiles -nDt "$TARGET" "$pkg"
    return
  fi

  stow --dotfiles -Dt "$TARGET" "$pkg"
  success "Unstowed $pkg"
}

migrate_home() {
  load_migrateignore

  info "Scanning home directory for dotfiles…"

  files=()
  while IFS= read -r file; do
    files+=("$file")
  done < <(detect_home_dotfiles)

  if [ ${#files[@]} -eq 0 ]; then
    warn "No dotfiles found to migrate"
    exit 0
  fi

  echo ""
  info "Migration plan:"
  i=1
  for file in "${files[@]}"; do
    base=$(basename "$file")
    clean="${base#.}"
    pkg="${clean%%.*}"
    echo "  [$i] $file → $DOTFILES_DIR/$pkg/dot-$clean"
    ((i++))
  done

  echo ""
  read -p "Select items to migrate (e.g. '1 2 5', 'all', or 'none'): " selection

  if [[ "$selection" == "none" ]]; then
    warn "No items selected. Aborting migration."
    exit 0
  fi

  selected_files=()

  if [[ "$selection" == "all" ]]; then
    selected_files=("${files[@]}")
  else
    for index in $selection; do
      if [[ "$index" =~ ^[0-9]+$ ]] && (( index >= 1 && index <= ${#files[@]} )); then
        selected_files+=("${files[$((index-1))]}")
      else
        warn "Invalid selection: $index"
      fi
    done
  fi

  echo ""
  info "Migrating selected items…"

  for file in "${selected_files[@]}"; do
    base=$(basename "$file")
    clean="${base#.}"
    pkg="${clean%%.*}"

    mkdir -p "$DOTFILES_DIR/$pkg"

    target="$DOTFILES_DIR/$pkg/dot-$clean"

    if [[ "$APPLY" != "1" ]]; then
      info "[DRY RUN] Would migrate $file → $target"
      continue
    fi

    backup_file "$file"
    mv "$file" "$target"
    ln -s "$target" "$file"

    success "Migrated $file → $target"
  done

  success "Migration complete"
}

main() {
  cd "$DOTFILES_DIR"

  if [[ "$MIGRATE" == "1" ]]; then
    migrate_home
    exit 0
  fi

  # Load packages into an array (portable version)
  packages=()
  while IFS= read -r pkg; do
    packages+=("$pkg")
  done < <(detect_packages)

  # Feedback when no packages exist
  if [ ${#packages[@]} -eq 0 ]; then
    warn "No packages found in $DOTFILES_DIR"
    info "Create a package directory (e.g., mkdir zshrc) and try again."
    exit 0
  fi

  if [[ "$LIST" == "1" ]]; then
    info "Detected packages:"
    printf '%s\n' "${packages[@]}"
    exit 0
  fi

  ensure_directories

  if [[ -n "$ONLY" ]]; then
    stow_package "$ONLY"
    exit 0
  fi

  for pkg in "${packages[@]}"; do
    if [[ "$UNINSTALL" == "1" ]]; then
      unstow_package "$pkg"
    else
      stow_package "$pkg"
    fi
  done

  success "All operations complete"
}

# Parse flags
while [[ $# -gt 0 ]]; do
  case "$1" in
    --apply) APPLY=1 ;;
    --force) FORCE=1 ;;
    --uninstall) UNINSTALL=1 ;;
    --adopt) ADOPT=1 ;;
    --verbose) VERBOSE=1 ;;
    --list) LIST=1 ;;
    --only) ONLY="$2"; shift ;;
    --migrate) MIGRATE=1 ;;
  esac
  shift
done

main

NAME
    bootstrap – portable dotfiles bootstrap and migration tool

SYNOPSIS
    bootstrap [--apply] [--migrate] [--only PKG]
              [--adopt] [--uninstall] [--force]
              [--list] [--verbose]

DESCRIPTION
    bootstrap is a portable, safety‑first dotfiles management tool
    built around GNU Stow. It treats the current working directory
    as the dotfiles root and provides interactive migration,
    automatic backups, and dry‑run behavior by default.
    
    bootstrap refuses to run if invoked from $HOME to prevent accidental
    migration or stowing of the entire home directory.

    No changes are made unless --apply is specified.

OPTIONS
    --apply
           Perform real actions. Without this flag, bootstrap
           previews all operations (dry‑run mode).

    --migrate
           Scan the user’s home directory for dotfiles, display a
           migration plan, and interactively select items to
           migrate into the dotfiles directory. Respects patterns
           in .migrateignore.

    --only PKG
           Operate only on the specified package. Useful for
           stowing or adopting a single component.

    --adopt
           Adopt existing files into a package. Stow will move
           files from the target directory into the package and
           replace them with symlinks.

    --uninstall
           Unstow packages, removing symlinks created by Stow.

    --force
           Automatically back up conflicting files before stowing
           or adopting. Backups are timestamped.

    --list

           A file in the dotfiles directory containing glob

               .cache/
               .local/**
               *.swp
               .zsh_sessions
