# Toggle the visibility of Volumes in the Viewport
import maya.cmds as cmds

if cmds.objExists("voodooProxyShape"):
    # Get the current visibility status
    vol_status = cmds.getAttr("voodooProxyShape.volumeDisplayEnabled")
    # Toggle the visibility status
    cmds.setAttr("voodooProxyShape.volumeDisplayEnabled", not vol_status)
