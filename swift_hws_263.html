import Foundation
import SwiftData

@Model
class Card: Identifiable {
    var id = UUID()
    var prompt: String
    var answer: String
    
    init(prompt: String, answer: String) {
        self.prompt = prompt
        self.answer = answer
    }
    
    init(card: Card) {
        prompt = card.prompt
        answer = card.answer
    }
    
    static let example = Card(prompt: "What is Swift?", answer: "A high-level programming language.")
}

CardView.swift

import SwiftUI

struct CardView: View {
    @Environment(\.accessibilityDifferentiateWithoutColor) var accessibilityDifferentiateWithoutColor
    @Environment(\.accessibilityVoiceOverEnabled) var accessibilityVoiceOverEnabled
    
    @State private var isShowingAnswer = false
    @State private var offset = CGSize.zero
    
    let card: Card
    var removal: ((Bool) -> Void)?
    
    var body: some View {
        ZStack {
            RoundedRectangle(cornerRadius: 25)
                .fill(
                    accessibilityDifferentiateWithoutColor
                    ? .white
                    : .white
                        .opacity(1 - Double(abs(offset.width / 50)))
                )
                .background(
                    accessibilityDifferentiateWithoutColor
                    ? nil
                    : RoundedRectangle(cornerRadius: 25)
                        .fill(offset.width > 0 ? .green : .red)
                )
                .shadow(radius: 10)
            
            VStack {
                if accessibilityVoiceOverEnabled {
                    Text(isShowingAnswer ? card.answer : card.prompt)
                        .font(.largeTitle)
                        .foregroundStyle(.black)
                } else {
                    Text(card.prompt)
                        .font(.largeTitle)
                        .foregroundStyle(.black)
                    
                    if isShowingAnswer {
                        Text(card.answer)
                            .font(.title)
                            .foregroundStyle(.secondary)
                    }
                }
            }
            .padding(20)
            .multilineTextAlignment(.center)
        }
        .frame(width: 450, height: 250)
        .rotationEffect(.degrees(Double(offset.width / 5)))
        .offset(x: offset.width * 5)
        .opacity(2 - Double(abs(offset.width / 50)))
        .accessibilityAddTraits(.isButton)
        .gesture(
            DragGesture()
                .onChanged { gesture in
                    offset = gesture.translation
                }
                .onEnded { _ in
                    if abs(offset.width) > 100 {
                        removal?(offset.width > 0)
                    } else {
                        offset = .zero
                    }
                }
        )
        .onTapGesture {
            isShowingAnswer.toggle()
        }
        .animation(.bouncy, value: offset)
    }
}

EditCards.swift

import SwiftData
import SwiftUI

struct EditCardsView: View {
    @Environment(\.dismiss) var dismiss
    @Environment(\.modelContext) var modelContext
    
    @Query private var cards: [Card]
    @State private var newPrompt = ""
    @State private var newAnswer = ""
    
    var body: some View {
        NavigationStack {
            List {
                Section("Add new card") {
                    TextField("Prompt", text: $newPrompt)
                    TextField("Answer", text: $newAnswer)
                    Button("Add Card", action: addCard)
                }
                
                Section {
                    ForEach(cards, id: \.id) { card in
                        VStack(alignment: .leading) {
                            Text(card.prompt)
                                .font(.headline)
                            Text(card.answer)
                                .foregroundStyle(.secondary)
                        }
                    }
                    .onDelete(perform: deleteCards)
                }
            }
            .navigationTitle("Edit Cards")
            .toolbar {
                Button("Done", action: done)
            }
        }
    }
    
    func done() {
        dismiss()
    }
    
    func addCard() {
        let trimmedPrompt = newPrompt.trimmingCharacters(in: .whitespaces)
        let trimmedAnswer = newAnswer.trimmingCharacters(in: .whitespaces)
        
        guard trimmedPrompt.isEmpty == false, trimmedAnswer.isEmpty == false else {
            return
        }
        
        let card = Card(prompt: trimmedPrompt, answer: trimmedAnswer)
        modelContext.insert(card)
        
        newPrompt = ""
        newAnswer = ""
    }
    
    func deleteCards(at offsets: IndexSet) {
        for offset in offsets {
            modelContext.delete(cards[offset])
        }
    }
}

import Combine
import SwiftData
import SwiftUI

extension View {
    func stacked(at position: Int, in total: Int) -> some View {
        let offset = Double(total - position)
        return self.offset(y: offset * 10)
    }
}

struct ContentView: View {
    @Environment(\.accessibilityDifferentiateWithoutColor) var accessibilityDifferentiateWithoutColor
    @Environment(\.accessibilityVoiceOverEnabled) var accessibilityVoiceOverEnabled
    @Environment(\.modelContext) var modelContext
    
    @Query private var cards: [Card]
    @State private var showingEditScreen = false
    
    @State private var timeRemaining = 100
    let timer = Timer.publish(every: 1, on: .main, in: .common).autoconnect()
    
    @Environment(\.scenePhase) var scenePhase
    @State private var isActive = true
    
    var body: some View {
        ZStack {
            Image(decorative: "background")
                .resizable()
                .ignoresSafeArea()
            
            VStack {
                Text("Time: \(timeRemaining)")
                    .font(.largeTitle)
                    .foregroundStyle(.white)
                    .padding(.horizontal, 20)
                    .padding(.vertical, 5)
                    .background(.black.opacity(0.75))
                    .clipShape(.capsule)
                
                ZStack {
                    ForEach(Array(cards.enumerated()), id: \.element.id) { index, card in
                        CardView(card: card) { isRight in
                            withAnimation {
                                removeCard(card, isRight: isRight)
                            }
                        }
                        .stacked(at: index, in: cards.count)
                        .allowsHitTesting(index == cards.count - 1)
                        .accessibilityHidden(index < cards.count - 1)
                    }
                }
                .allowsHitTesting(timeRemaining > 0)
                
                if cards.isEmpty {
                    Button("Start Again", action: resetCards)
                        .padding()
                        .background(.white)
                        .foregroundStyle(.black)
                        .clipShape(.capsule)
                }
            }
            
            VStack {
                HStack {
                    Spacer()
                    
                    Button {
                        showingEditScreen = true
                    } label: {
                        Image(systemName: "plus.circle")
                            .padding()
                            .background(.black.opacity(0.7))
                            .clipShape(.circle)
                    }
                }
                
                Spacer()
            }
            .foregroundStyle(.white)
            .font(.largeTitle)
            .padding()
            
            if accessibilityDifferentiateWithoutColor || accessibilityVoiceOverEnabled {
                VStack {
                    Spacer()
                    
                    HStack {
                        Button {
                            withAnimation {
                                removeCard(cards.last, isRight: false)
                            }
                        } label: {
                            Image(systemName: "xmark.circle")
                                .padding()
                                .background(.black.opacity(0.7))
                                .clipShape(.circle)
                        }
                        .accessibilityLabel("Wrong")
                        .accessibilityHint("Mark your answer as being incorrect.")
                        
                        Spacer()
                        
                        Button {
                            withAnimation {
                                removeCard(cards.last, isRight: true)
                            }
                        } label: {
                            Image(systemName: "checkmark.circle")
                                .padding()
                                .background(.black.opacity(0.7))
                                .clipShape(.circle)
                        }
                        .accessibilityLabel("Correct")
                        .accessibilityHint("Mark your answer as being correct.")
                    }
                    .foregroundStyle(.white)
                    .font(.largeTitle)
                    .padding()
                }
            }
        }
        .onReceive(timer) { _ in
            guard isActive else { return }
            
            if timeRemaining > 0 {
                timeRemaining -= 1
            }
        }
        .onChange(of: scenePhase) {
            if scenePhase == .active {
                if cards.isEmpty == false {
                    isActive = true
                }
            } else {
                isActive = false
            }
        }
        .sheet(isPresented: $showingEditScreen, onDismiss: resetCards) {
            EditCardsView()
        }        .onAppear(perform: resetCards)
    }
    
    func removeCard(_ card: Card?, isRight: Bool) {
        guard let card else { return }
        
        modelContext.delete(card)
        if !isRight {
            modelContext.insert(Card(card: card))
        }
        
        if cards.isEmpty {
            isActive = false
        }
    }
    
    func resetCards() {
        timeRemaining = 100
        isActive = true
    }
}

Ai Insight on cdalvaro's solution

Challenge 3 is definitely where things get a bit mind-bending because it introduces a complete architecture shift.

Looking closely at your provided code, you've actually hit two major birds with one stone here: migrating from UserDefaults to SwiftData and fixing the unique identifier constraint.

Here is a breakdown of exactly what changed and how it satisfies the goal.

1. Why ForEach(0..<cards.count, id: \.self) Had to Go

In your original code, the ForEach loop relied on array indices to render the card stack: 

ForEach(0..<cards.count, id: \.self) { index in ... }

When you use id: \.self with integers, SwiftUI binds the identity of the view to its position in the stack rather than the actual card data.

The Bug: If you answer a card incorrectly and shove it to the bottom of the array, the view at index doesn't change smoothly. SwiftUI gets confused because the indices shift around, causing animation glitches (like cards instantly teleporting or flashing back instead of sliding gracefully).

The Solution in ContentView.swift:

ForEach(Array(cards.enumerated()), id: \.element.id) { index, card in ... }

By switching to an enumerated array and setting id: \.element.id, you give SwiftUI a permanent anchor. Even if a card is deleted, re-inserted, or rearranged, SwiftUI tracks it by its unique identity (id), ensuring perfect, fluid card animations.

2. The Move to SwiftData (The Magic of Re-Insertion)

To dynamically insert cards back into the persistent stack without messy manual JSON re-encoding, the project shifted to using Apple's modern SwiftData framework.

The Card.swift file's new shape now sports the @Model macro and looks like this: 

import Foundation
import SwiftData

@Model
class Card: Identifiable {
    var id = UUID()
    var prompt: String
    var answer: String
    
    init(prompt: String, answer: String) {
        self.prompt = prompt
        self.answer = answer
    }
    
    init(card: Card) {
        prompt = card.prompt
        answer = card.answer
    }
    
    static let example = Card(prompt: "What is Swift?", answer: "A high-level programming language.")
}

Because cards is now managed globally via @Query private var cards: [Card], anytime you touch the modelContext, the UI updates automatically.

3. How the "Wrong Answer" Logic Actually Works

Look closely at how your removeCard function changed in ContentView.swift:

Old Implementation:

func removeCard(at index: Int) {
    cards.remove(at: index)
}

It didn't know (or care) if you swiped left or right; the card was permanently vaporized from the array.

New Implementation:

func removeCard(_ card: Card?, isRight: Bool) {
    guard let card else { return }
    
    modelContext.delete(card) // 1. Always remove the active card instance
    
    if !isRight {
        modelContext.insert(Card(card: card)) // 2. Put a fresh clone back into the stack
    }
    ...
}

Breaking down the new removeCard:

  • removal?(offset.width > 0): In CardView.swift, when a user finishes a drag gesture, it checks the swipe direction. A right swipe (offset.width > 0) passes true (Correct). A left swipe passes false (Wrong).
  • modelContext.delete(card): No matter what, the specific card you just swiped is removed from the current context so it flies off the screen.
  • modelContext.insert(Card(card: card)): If isRight is false (you got it wrong), a brand new clone of the card is inserted into SwiftData.

Because it’s a new instance, it gets a fresh UUID. SwiftUI treats it as a completely new view entering the deck rather than a recycled index, meaning it drops cleanly back down to the bottom of the pile without breaking your existing stack visuals.

Your Card model holds the exact secret to why the re-insertion trick works without breaking SwiftUI's animations. Let’s break down the two most important parts of this model file.

1. The Power of var id = UUID()

By adding var id = UUID() inside a SwiftData class, every single card created gets its own universally unique identifier.

When your ContentView loops through the cards using: 

ForEach(Array(cards.enumerated()), id: \.element.id)

SwiftUI isn't looking at the content (the text) or the index (position 0, 1, 2). It looks explicitly at that id.

If you had relied on the old id: \.self with integers or strings, duplicating a card would cause a massive collision. SwiftUI would see two views with the exact same identity, panic, and either crash or fail to animate. Because every card now has a totally distinct id, SwiftUI treats them as completely separate physical objects in the virtual stack.

QUESTION

What is the 'element' in: 

ForEach(Array(cards.enumerated()), id: \.element.id)

...Why not just .id?

ANSWER

That is a fantastic question. It hits right at the heart of how Swift’s type system transforms data when you change how you loop over it.

The short answer is: .element is required because enumerated() wraps your cards inside a special "tuple" (a paired pair of values), changing the structure of what ForEach is looking at.

Here is the exact breakdown of why .id alone would cause a compiler error, and why .element saves the day.

1. What enumerated() Does to Your Array

Normally, cards is a straight list of Card objects: [Card]. If you looped over it directly, each item would just be a Card.

However, to get the visual stacking math right (.stacked(at: index, in: cards.count)), you need both the card itself AND its index number (its position in the array). To get both at the same time, you use cards.enumerated().

When you call .enumerated(), Swift transforms each single item into a paired structure called an EnumeratedSequence Element. This structure contains exactly two properties:

  • .offset: The integer index of the item (0, 1, 2, etc.).
  • .element: The actual underlying data object (your Card).

To visualize Array(cards.enumerated()), imagine a storage box that contains smaller, labeled boxes.

Originally, your array is just a flat line of Card objects. But when you wrap it in Array(cards.enumerated()), Swift repackages each item into a compound structure (a tuple).

Here is exactly what that data structure looks like conceptually inside your app's memory: 

[
  // Box 0
  (
    offset: 0,
    element: Card(id: UUID("A1-B2..."), prompt: "What is Swift?", answer: "...")
  ),
  
  // Box 1
  (
    offset: 1,
    element: Card(id: UUID("C3-D4..."), prompt: "Who played the 13th Doctor?", answer: "...")
  ),
  
  // Box 2
  (
    offset: 2,
    element: Card(id: UUID("E5-F6..."), prompt: "What is SwiftData?", answer: "...")
  )
]

Navigating the Structure

When you write your ForEach loop, you are interacting with this nested layout in two distinct ways:

  • Identifying the item (id: \.element.id): SwiftUI looks at the outer array item, enters the .element box to find your Card, and then reads the .id property inside it.
  • Using the item ({ index, card in ... }): Swift automatically unpacks the tuple for you. It takes the .offset value and assigns it to your index variable, and takes the .element value and assigns it to your card variable.

This structural blueprint allows your code to simultaneously know a card's exact physical order in the deck (via index) while securely tracking its true, unchanging identity (via card.id).

2. Why Can't You Just Use id: \.id?

When ForEach looks at Array(cards.enumerated()), it isn't looking at a list of Card structs anymore. It is looking at a list of these paired (offset: Int, element: Card) structures.

If you write id: \.id, the Swift compiler searches the outer wrapper for a property named id. It looks at the wrapper and says: "Hey, I see an .offset property, and I see an .element property... but I don't see any property called .id out here!"

Because .id belongs to your Card class, not the enumeration wrapper, the compiler panics and throws an error.

3. Why \.element.id Works Perfectly

By writing id: \.element.id, you are giving SwiftUI a precise roadmap to dig through the wrapper:

  • \.element: Tells SwiftUI to step past the outer wrapper and look directly at the underlying Card object.
  • .id: Tells SwiftUI to grab the unique UUID sitting inside that card.

This gives ForEach the permanent, unique identifier it needs to track the card across swipes and re-insertions, while still allowing the closure to hand you the index (from .offset) to handle your stacking layout.

2. The Cloning Blueprint: init(card: Card)

This initializer is a brilliant, clean way to handle the "wrong answer" behavior: 

init(card: Card) {
    prompt = card.prompt
    answer = card.answer
}

Notice what is missing from this initializer? It doesn't copy card.id.

Because id has a default value (var id = UUID()), whenever you call this specific initializer, Swift automatically generates a brand-new, completely fresh UUID for the cloned card, while copying over the old prompt and answer.

Putting it into the App's Flow:

  • You swipe left on a card because you got it wrong.
  • removeCard calls modelContext.delete(card). The old card - along with its unique ID - is completely removed from SwiftData and the screen.
  • modelContext.insert(Card(card: card)) runs. This calls your cloning initializer.
  • SwiftData receives a brand-new card object with identical text but a completely new ID.

To SwiftUI, it doesn't see a card "moving backward" in the stack. It sees an entirely new object arriving at the bottom of the pile. This allows SwiftUI to effortlessly animate the old card flying away, while gracefully keeping the new copy waiting at the bottom of the deck for your next attempt!

3. id: \.element.id (What is it doing here?)

This is the key to the whole operation. This tells SwiftUI how to recognize each item in the array so it can track them over time.

Whenever you use a ForEach loop to build views dynamically, SwiftUI demands to know: "If the data changes, how do I know which view belongs to which piece of data?"

If you don't provide a unique identifier, SwiftUI won't know the difference between a card being swiped away and a card simply changing its text.

By writing id: \.element.id, you are telling SwiftUI:

"Hey SwiftUI, as you loop through this array of boxes, I want you to look inside the .element variable (which is the Card), find its .id property (the unique UUID), and use that string as the permanent passport number for this view."

The Big Picture: Why this matters for Challenge 3

Think about what happens when you swipe a card wrong:

  • The old card gets deleted.
  • A cloned card with a new UUID is added to the bottom of the array.
  • ForEach reruns.

Because of id: \.element.id, SwiftUI looks at the updated array and says:

  • "I see the top card's UUID is completely gone. I will animate it flying off the screen."
  • "I see a completely new UUID at the bottom of the array. I will smoothly animate a brand-new card appearing at the bottom of the stack."

If you had used the old id: \.self with integers, SwiftUI would only look at the positions (0, 1, 2). It would see that the array size is still the same and say, "Oh, nothing moved, the slots are the same!" - causing the old card to instantly snap back to the center of the screen instead of flying away.
