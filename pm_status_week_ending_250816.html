AutoCrop button

I made a script node to turn on and off autocrop nodes.

  • s 2230 0350:
    • Without AutoCrops:
      • 20 sec
    • With AutoCrops:
      • 11 sec

Added this to a NoOp (Manage User Knobs... Python Script Buttons x2): 

import nuke

def autocrops(node_list, switch=1):
    nodes = []
    for node in node_list:
        cls = node.Class()
        if cls == "Crop":
            
            if node.knob("label").value() == "AutoCrop":
                node.knob("disable").clearAnimated()
                node.knob("disable").setValue(switch)

            if node.knob("label").value() == "AutoCrop":
                node.knob("disable").setValue(switch)
                nodes.append(node["name"].value())

        elif cls == "Group":
            try:
                autocrops(node.nodes(), switch)
            except:
                pass

# autocrops(node_list, 0) # ON
autocrops(node_list, 1) # OFF
