#! /usr/bin/env python3

from PySide6.QtCore import QTimer
from PySide6.QtWidgets import QApplication, QWidget, QPushButton, QVBoxLayout

class MyWidget(QWidget):
    def __init__(self):
        super().__init__()

        # Initiate a timer:
        self.timer = QTimer(self)
        # At each timer (repeated), activate a method:
        self.timer.timeout.connect(self.update)

        self.button = QPushButton("Start Timer", self)
        # Connect button to a custom method
        self.button.clicked.connect(self.toggle_timer)

        layout = QVBoxLayout(self)
        layout.addWidget(self.button)

        self.setLayout(layout)

    def toggle_timer(self):
    """ Method to activate and deactivate the QTimer (via a button).
    Arg:
        None
    Returns:
        None
    """
        # Check to see if the QTimer is counting, and toggle on and off:
        if self.timer.isActive():
            self.timer.stop()
            self.button.setText("Start Timer")
        else:
            self.timer.start(1000)  # Timer will timeout every 1000 milliseconds (1 second)
            self.button.setText("Stop Timer")

    def update(self):
    """ Arbitrary example of an action to perform when timer dictates.
    Arg:
        None
    Returns:
        None
    """
        print("Timer timeout!")

app = QApplication([])
widget = MyWidget()
widget.show()
app.exec()
