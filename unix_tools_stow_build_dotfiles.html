#!/usr/bin/env bash
set -e

DOTFILES_DIR="$HOME/dotfiles"
TARGET="$HOME"

# Flags
FORCE=0
UNINSTALL=0
DRYRUN=1
ADOPT=0
VERBOSE=0
LIST=0
ONLY=""

# Colors
RED="\033[0;31m"
GREEN="\033[0;32m"
YELLOW="\033[1;33m"
BLUE="\033[0;34m"
RESET="\033[0m"

info()    { echo -e "${BLUE}INFO:${RESET} $1"; }
warn()    { echo -e "${YELLOW}WARN:${RESET} $1"; }
error()   { echo -e "${RED}ERROR:${RESET} $1"; }
success() { echo -e "${GREEN}OK:${RESET} $1"; }

# Cross-platform package detection (no GNU find)
detect_packages() {
  for dir in "$DOTFILES_DIR"/*; do
    # Only treat as a package if it's a directory AND not the .git folder
    if [[ -d "$dir" && "$(basename "$dir")" != ".git" ]]; then
      basename "$dir"
    fi
  done
}

check_conflicts() {
  local pkg="$1"
  stow --dotfiles -nvt "$TARGET" "$pkg" 2>&1 | grep "existing target" || true
}

backup_file() {
  local file="$1"
  local backup="$file.backup.$(date +%Y%m%d%H%M%S)"
  warn "Backing up $file → $backup"
  mv "$file" "$backup"
}

ensure_directories() {
  info "Ensuring required directories exist"
  mkdir -p "$HOME/.config"
  mkdir -p "$HOME/.local/share"
}

stow_package() {
  local pkg="$1"
  info "Processing package: $pkg"

  # 1. Check for Conflicts
  local conflicts
  conflicts=$(check_conflicts "$pkg")

  if [[ -n "$conflicts" ]]; then
    if [[ "$FORCE" == "1" ]]; then
      echo "$conflicts" | sed 's/.*existing target: //' | while read -r file; do
        # Use our safe dry-run backup function
        backup_file "$TARGET/$file"
      done
    else
      warn "Conflicts detected for $pkg. Run with --force to see preview."
      return
    fi
  fi

  # 2. THE FIX: Run Stow with --restow (-R) in dry-run mode to "see through" old links
  # -R tells Stow: "Ignore existing links and show me what the NEW ones would be."
  local STOW_OPTS="-vvt $TARGET --dotfiles"
  [[ "$DRYRUN" == "1" ]] && STOW_OPTS="-n $STOW_OPTS"
  
  # Use -R (Restow) here so it calculates the arrows even if old links exist
  local cmd="stow $STOW_OPTS -R $pkg"
  
  $cmd 2>&1 | sed -n 's/^LINK: \(.*\) => \(.*\)/\1 --> \2/p'
}

unstow_package() {
  local pkg="$1"
  info "Unstowing $pkg"
  stow --dotfiles -Dt "$TARGET" "$pkg"
  success "Unstowed $pkg"
}

main() {
  if [[ "$DRYRUN" == "1" ]]; then
    echo -e "${YELLOW}------------------------------------------------------------${RESET}"
    echo -e "${YELLOW}  RUNNING IN SIMULATION ONLY - Run with --apply to execute${RESET}"
    echo -e "${YELLOW}------------------------------------------------------------${RESET}"
    echo ""
  fi
  cd "$DOTFILES_DIR"

  local packages
  packages=$(detect_packages)

  if [[ "$LIST" == "1" ]]; then
    info "Detected packages:"
    echo "$packages"
    exit 0
  fi

  ensure_directories

  if [[ -n "$ONLY" ]]; then
    stow_package "$ONLY"
    exit 0
  fi

  for pkg in $packages; do
    if [[ "$UNINSTALL" == "1" ]]; then
      unstow_package "$pkg"
    else
      stow_package "$pkg"
    fi
  done

  success "All operations complete"
}

# Parse flags
while [[ $# -gt 0 ]]; do
  case "$1" in
    --force) FORCE=1 ;;
    --uninstall) UNINSTALL=1 ;;
    --apply) DRYRUN=0 ;;
    --dry-run) DRYRUN=1 ;;
    --adopt) ADOPT=1 ;;
    --verbose) VERBOSE=1 ;;
    --list) LIST=1 ;;
    --only) ONLY="$2"; shift ;;
  esac
  shift
done

main

# Dotty — A Friendly Dotfiles Bootstrapper

Dotty is a Bash-based dotfiles bootstrap tool built on top of GNU Stow.  
It provides a clean, interactive workflow for managing dotfiles using simple
package directories such as:

dotty/
zshrc/
vim/
git/

Dotty automatically:

- Infers which `$HOME` files belong to each package (prefix-based)
- Shows a numbered selection list for each package
- Auto-includes existing dotfiles already inside the package
- Previews all symlinks in a grouped, human-friendly format
- Migrates selected files into the package (with backups)
- Symlinks them back into `$HOME`
- Uses Stow to manage the final structure

No more manual renaming to `dot-*`.  
No more guessing what Stow will do.  
No more separate “migrate” mode — migration is built in.

---

## 🚀 Quick Start

```bash
git clone <your repo> ~/dotty
cd ~/dotty
./bootstrap.sh
```

Dotty will:
- Detect each package directory
- Infer matching $HOME files (prefix-based)
- Show a selection UI
- Preview the symlink plan
- Apply changes only when you use --apply

---

# Dotty Cheat Sheet

## Basic Usage
- Dry run (safe):  
  `./bootstrap.sh`

- Apply changes:  
  `./bootstrap.sh --apply`

- Uninstall all packages:  
  `./bootstrap.sh --uninstall`

- Adopt unmanaged files:  
  `./bootstrap.sh --adopt`

- List packages:  
  `./bootstrap.sh --list`

- Process only one package:  
  `./bootstrap.sh --only zshrc`

---

## Workflow Summary
1. Dotty detects each package directory.
2. Shows existing dotfiles (auto-included).
3. Infers `$HOME` files by prefix.
4. Shows numbered selection list.
5. Dry run preview (grouped).
6. Apply mode migrates + symlinks + stows.

---

## Selection UI
- `all` → include all inferred files  
- `none` → include none  
- `1 3` → include specific files  
- **Enter** → default = none  

---

## Package Naming Rules
Package `zshrc` matches:

- `~/.zshrc`
- `~/.zshenv`
- `~/.zprofile`
- `~/.zlogin`

Package `vim` matches:

- `~/.vimrc`
- `~/.vim/`

---

## Safety
- Backups created automatically
- Dry run by default
- No destructive actions without `--apply`

## 🧠 How Dotty Infers Files

A package named:

```sh
zshrc
```
will match any $HOME file whose name begins with:

```sh
.zsh*
```
Examples:
- ~/.zshrc
- ~/.zshenv
- ~/.zprofile
- ~/.zlogin

This is Option A (prefix-based inference).

## 🧩 Selection UI

For each package, Dotty shows:
1. Existing dotfiles in the package (auto-included)

```sh
Existing dotfiles in package (auto-included):
  dot-zshrc
  dot-zshenv
```

2. Inferred $HOME files (user-selectable)

```sh
Home files matching this package:

  [1] ~/.zshrc
  [2] ~/.zshenv
  [3] ~/.zprofile

Select items to include (e.g. "1 3", "all", or "none"; default: none):
```

3. Default behavior
- Pressing Enter = none
- all = include all inferred files
- none = include none
- 1 3 = include only those

## 🧪 Dry Run Mode (default)

Dotty never modifies your system unless you use --apply.

Dry run preview example:

```sh
[DRY RUN] Would stow zshrc

Existing dotfiles in package (auto-included):
  dot-zshrc

New symlinks to create:
  ~/.zshenv → dotty/zshrc/dot-zshenv
  ~/.zprofile → dotty/zshrc/dot-zshprofile
```

## 🔧 Apply Mode

To actually migrate and symlink:

```bash
./bootstrap.sh --apply
```

Dotty will:
- Backup each selected $HOME file
- Move it into the package as dot-*
- Create the symlink
- Run Stow to finalize the structure

📙 MAN PAGE (bootstrap.1)

In Linux and Unix-like systems, a roff-formatted man page should be saved in a specific subdirectory within the system's manual path so the man command can find and display it.

1. Where to Save It

Since your text describes a Bash shell script (an executable command), it belongs in Section 1 of the manual.

  • System-wide (Standard): Save to /usr/local/share/man/man1/.
    • Note: Using /usr/local is preferred over /usr/share for manually added files to avoid conflicts with your package manager.
  • User-only (Personal): Save to ~/.local/share/man/man1/ or ~/man/man1/.
    • If you use a custom path like ~/man, you must add it to your environment by putting export MANPATH="\(HOME/man:\)MANPATH" in your .bashrc.

2. How to Save and Install It

Follow these steps to correctly name and "register" your man page:

  1. Name the file: The filename must end with the section number. If your script is named myscript, name the man page myscript.1.
  2. Move the file: 
sudo cp myscript.1 /usr/local/share/man/man1/

Compress it (Optional but standard): Most systems expect man pages to be gzipped to save space. 

sudo gzip /usr/local/share/man/man1/myscript.1

Update the database: Run mandb to ensure the system recognizes the new entry immediately. 

sudo mandb

3. Quick Verification

To test if it's working without moving files yet, you can view the local file directly: 

man ./myscript.1

Once installed, you can simply run: 

man myscript

.TH BOOTSTRAP 1 "February 2024" "1.0" "Dotfiles Management Manual"
.SH NAME
bootstrap \- automated dotfiles symlinking and management using GNU Stow
.SH SYNOPSIS
.B ./bootstrap.sh
[\fIOPTIONS\fR]
.SH DESCRIPTION
.B bootstrap.sh
is a wrapper script for GNU Stow designed to organize and deploy configuration files (dotfiles). 
It automates the process of creating symlinks from a central repository (default: ~/dotfiles) 
to the home directory, while providing safety features like dry-runs and automatic backups.
.PP
By default, the script runs in \fBSIMULATION ONLY\fR mode. No changes are made to the 
filesystem unless the \fB\-\-apply\fR flag is provided.
.SH OPTIONS
.TP
.B \-\-apply
Execute the operations. This is required to make actual changes to the filesystem.
.TP
.B \-\-force
Enable force mode. If a conflict is detected with an existing file that is not a symlink 
or points elsewhere, the script will automatically back up the original file and replace it.
.TP
.B \-\-dry\-run
Explicitly run in simulation mode. This is the default behavior.
.TP
.B \-\-adopt
Import existing files into the dotfiles repository. WARNING: This will overwrite 
files in your repository with the versions currently in your home directory.
.TP
.B \-\-uninstall
Remove symlinks created by Stow for all detected packages.
.TP
.B \-\-list
List all detected packages in the dotfiles directory and exit.
.TP
.B \-\-only \fIPACKAGE\fR
Process only the specified package instead of all detected packages.
.SH FILES
.TP
.I ~/dotfiles
The default directory where configuration packages are stored.
.TP
.I ~/.config
The script ensures this directory exists before processing.
.SH EXAMPLES
.TP
.B ./bootstrap.sh
Run a safe simulation to see what links would be created.
.TP
.B ./bootstrap.sh \-\-force
Preview which files would be backed up and replaced.
.TP
.B ./bootstrap.sh \-\-apply \-\-only zsh
Apply only the 'zsh' configuration package.
.SH AUTHOR
Created for custom dotfiles management.
.SH SEE ALSO
.BR stow (1)
