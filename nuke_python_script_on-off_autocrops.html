NoOp Buttons - ON/OFF AutoCrops

This script features a recursive feature for delving into Groups to look for more AutoCrops nodes.:w

  • Create a NoOp node
  • Right-click:
    • 'Manage User Knobs...'
    • Choose Add > 'Python Script Button'
    • name, label, then add the script below:
      • Uncomment on or off lines
    • Create second 'Python Script Button':
      • name, label, then add the script below:
      • Uncomment on or off lines 
        import nuke

def autocrops(node_list, switch=1):
    nodes = []
    for node in node_list:
        cls = node.Class()
        if cls == "Crop":
            
            if node.knob("label").value() == "AutoCrop":
                node.knob("disable").clearAnimated()
                node.knob("disable").setValue(switch)

            if node.knob("label").value() == "AutoCrop":
                node.knob("disable").setValue(switch)
                nodes.append(node["name"].value())

        elif cls == "Group":
            try:
                autocrops(node.nodes(), switch)
            except:
                pass
                
    for n in nodes:
        status = "OFF" if switch else "ON"
        print(f"{n} has been switch {status}")
