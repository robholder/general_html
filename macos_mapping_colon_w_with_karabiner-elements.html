MacOS - Mapping ':w' to 'save' in Xcode with Karabiner-Elements

Install Karabiner-Elements, and allow all the premissions it asks for from the System.

Xcode - ':w' to 'save'

Xcode has a basic Vim emulation option. Editor > Vim Mode.

This mod makes :w (the Vim 'save buffer' command) trigger the <Cmd> s save shortcut in Xcode.

  • Add this to the Karabiner-Elements Settings > 'Complex Modifications': 
{
    "description": "Map :w to Cmd+S in Xcode",
    "manipulators": [
        {
            "conditions": [
                {
                    "bundle_identifiers": [
                        "^com\\.apple\\.dt\\.Xcode$"
                    ],
                    "type": "frontmost_application_if"
                },
                {
                    "name": "colon_pressed",
                    "type": "variable_if",
                    "value": 1
                }
            ],
            "from": { "key_code": "w" },
            "to": [
                { "key_code": "delete_or_backspace" },
                {
                    "key_code": "s",
                    "modifiers": ["left_command"]
                }
            ],
            "type": "basic"
        },
        {
            "conditions": [
                {
                    "bundle_identifiers": [
                        "^com\\.apple\\.dt\\.Xcode$"
                    ],
                    "type": "frontmost_application_if"
                }
            ],
            "from": {
                "key_code": "semicolon",
                "modifiers": { "mandatory": ["left_shift"] }
            },
            "parameters": { "basic.to_delayed_action_delay_milliseconds": 500 },
            "to": [
                {
                    "set_variable": {
                        "name": "colon_pressed",
                        "value": 1
                    }
                },
                {
                    "key_code": "semicolon",
                    "modifiers": ["left_shift"]
                }
            ],
            "to_delayed_action": {
                "to_if_canceled": [
                    {
                        "set_variable": {
                            "name": "colon_pressed",
                            "value": 0
                        }
                    }
                ],
                "to_if_invoked": [
                    {
                        "set_variable": {
                            "name": "colon_pressed",
                            "value": 0
                        }
                    }
                ]
            },
            "type": "basic"
        }
    ]
}
