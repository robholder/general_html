Python - CSV Splitter

Splitting CSVs by Month

Here is a complete Python script to handle this for you. It uses Python's built-in csv and datetime modules, so you don't need to install any external libraries (like pandas).

The script reads your master CSV file, looks at the Posting Date for each row, figures out the month and year, and automatically groups and saves them into individual monthly files with the original headers intact.

The Python Script

import csv
from datetime import datetime
import os

# --- CONFIGURATION ---
# Replace 'master_bank_file.csv' with the actual path to your downloaded file
INPUT_FILE = 'master_bank_file.csv'
# The directory where you want the monthly files to be saved (current directory by default)
OUTPUT_DIR = 'monthly_statements'
# ---------------------

def split_csv_by_month(input_file, output_dir):
    # Create the output directory if it doesn't exist
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)
        print(f"Created directory: {output_dir}")

    # Dictionary to hold file handles and csv.writer objects for each month
    # Structure: {(year, month): (file_object, csv_writer_object)}
    monthly_writers = {}

    try:
        with open(input_file, mode='r', newline='', encoding='utf-8') as infile:
            # Use DictReader so we can access columns by their heading names
            reader = csv.DictReader(infile)
            headers = reader.fieldnames
            
            if not headers:
                print("Error: The input CSV file is empty or missing headers.")
                return

            print("Processing rows...")
            row_count = 0

            for row in reader:
                # 1. Extract the Posting Date
                date_str = row.get('Posting Date')
                if not date_str:
                    continue  # Skip rows with missing dates
                
                # 2. Parse the date to extract year and month
                # NOTE: Adjust the format '%m/%d/%Y' if your bank uses a different format 
                # (e.g., '%Y-%m-%d' or '%d/%m/%Y')
                try:
                    # Trying standard MM/DD/YYYY first
                    date_obj = datetime.strptime(date_str.strip(), '%m/%d/%Y')
                except ValueError:
                    try:
                        # Fallback for YYYY-MM-DD just in case
                        date_obj = datetime.strptime(date_str.strip(), '%Y-%m-%d')
                    except ValueError:
                        print(f"Skipping row due to unexpected date format: {date_str}")
                        continue

                year_month_key = (date_obj.year, date_obj.month)

                # 3. If this is a new month, create a new file and write the headers
                if year_month_key not in monthly_writers:
                    file_name = f"Statement_{date_obj.year}_{date_obj.month:02d}.csv"
                    file_path = os.path.join(output_dir, file_name)
                    
                    # Open the new file
                    outfile = open(file_path, mode='w', newline='', encoding='utf-8')
                    writer = csv.DictWriter(outfile, fieldnames=headers)
                    writer.writeheader()  # Re-instate headings
                    
                    # Store the file handle and writer so we can keep using them
                    monthly_writers[year_month_key] = (outfile, writer)
                
                # 4. Write the row to the appropriate monthly file
                _, writer = monthly_writers[year_month_key]
                writer.writerow(row)
                row_count += 1

            print(f"Success! Processed {row_count} rows into {len(monthly_writers)} monthly files.")

    except FileNotFoundError:
        print(f"Error: The file '{input_file}' was not found. Please check the path.")
    finally:
        # Clean up: Close all open output files safely
        for outfile, _ in monthly_writers.values():
            outfile.close()

if __name__ == "__main__":
    split_csv_by_month(INPUT_FILE, OUTPUT_DIR)

How to use it:

  • Save the Script: Copy the code above and save it as split_bank_csv.py.
  • Place your CSV: Put your large, multi-month bank CSV file in the same folder as the script, and make sure its filename matches the INPUT_FILE variable in the script (currently set to 'master_bank_file.csv').
  • Run the Script: Open your terminal or command prompt, navigate to the folder, and run: 
python split_bank_csv.py
  • Check the Output: A new folder called monthly_statements will appear, containing files named like Statement_2026_01.csv, Statement_2026_02.csv, etc., each complete with your original headings.

⚠️ One Quick Thing to Double-Check

Banks format dates differently depending on where you live and who you bank with. In the script, I used the standard US format: MM/DD/YYYY (e.g., 05/23/2026).

If your bank formats dates differently—for instance, DD/MM/YYYY - you'll just need to tweak line 39 in the script: For DD/MM/YYYY (e.g., 23/05/2026), change '%m/%d/%Y' to '%d/%m/%Y'.

    # Set up argument parsing
    parser = argparse.ArgumentParser(
        description="Split a multi-month bank CSV file into individual monthly files."
    )
    # This makes the filename a required positional argument
    parser.add_argument(
        'filename', 
        help="The path to the bank CSV file you want to split."
    )
    # This adds an optional argument for the output directory, defaulting to 'monthly_statements'
    parser.add_argument(
        '-o', '--output', 
        default='monthly_statements',
        help="The directory to save the split files (default: 'monthly_statements')"
    )
    
    args = parser.parse_args()

    # Run the function with the arguments supplied by the user
    split_csv_by_month(args.filename, args.output)

How to run it now:

You can pass the filename directly after the script name.

Basic usage:

python split_bank_csv.py my_bank_export.csv

If you want to save the output to a custom folder name:

  • I also added an optional -o flag in case you want to change the target folder on the fly: 
python split_bank_csv.py my_bank_export.csv -o Q1_Statements
