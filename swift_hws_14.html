Hacking With Swift - 14 - How to check multiple conditions

Multiple Conditions

Inefficient - redundant second 'test': 

let age = 16

if age >= 18 {
    print("You can vote in the next election.")
}

if age < 18 {
    print("Sorry, you're too young to vote.")
}

More efficient: 

let age = 16

if age >= 18 {
    print("You can vote in the next election.")
} else {
    print("Sorry, you're too young to vote.")
}

Else if 

if a {
    print("Code to run if 'a' is true")
} else if b {
    print("Code to run if 'a' is false, but 'b' is true")
} else {
    print("Code to run if both 'a' and 'b' are false")
}

Other multiple possibilities:

'AND' Nested:

let temp = 25

if temp > 20 {
    if temp < 30 {
        print("It's a nice day.")
    }
}

'AND' && Better than nested in this case:

let temp = 25

if temp > 20 && temp < 30 {
    print("It's a nice day.")
}
